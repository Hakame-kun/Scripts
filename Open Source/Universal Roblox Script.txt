-- UNDERGROUND RECOVERY
-- =====================================================================
local undergroundRecoveryConn = nil
local RECOVERY_CHECK_INTERVAL = 0.1
local GROUND_CHECK_DISTANCE   = 1000
local GROUND_BUFFER           = 3.5

local function isAnySpeedFeatureActive()
    return flyActive or vehicleFlyActive or vehiclespeedActive or toggles.walkspeed
end

local _undergroundRayParams        = RaycastParams.new()
_undergroundRayParams.FilterType   = Enum.RaycastFilterType.Exclude
local _undergroundFilterChar       = nil
local _UNDERGROUND_ORIGIN_OFFSET   = Vector3.new(0, 1, 0)
local _UNDERGROUND_RAY_DIR         = Vector3.new(0, -GROUND_CHECK_DISTANCE, 0)

local function tryRecoverFromUnderground()
    local char = localPlayer.Character
    if not char then return end
    local rootPart = char:FindFirstChild("HumanoidRootPart")
    if not rootPart then return end
    if _undergroundFilterChar ~= char then
        _undergroundFilterChar = char
        _undergroundRayParams.FilterDescendantsInstances = { char }
    end
    local pos    = rootPart.Position
    local result = workspace:Raycast(pos + _UNDERGROUND_ORIGIN_OFFSET, _UNDERGROUND_RAY_DIR, _undergroundRayParams)
    if result then
        local groundY = result.Position.Y
        if pos.Y < groundY - 1 then
            rootPart.CFrame = CFrame.new(pos.X, groundY + GROUND_BUFFER, pos.Z)
        end
    end
end

local function startUndergroundRecovery()
    if undergroundRecoveryConn then return end
    local lastCheck = 0
    undergroundRecoveryConn = RunService.Heartbeat:Connect(function()
        if not isAnySpeedFeatureActive() then return end
        local now = tick()
        if now - lastCheck < RECOVERY_CHECK_INTERVAL then return end
        lastCheck = now
        tryRecoverFromUnderground()
    end)
end

local function stopUndergroundRecovery()
    if undergroundRecoveryConn then
        undergroundRecoveryConn:Disconnect()
        undergroundRecoveryConn = nil
    end
    -- Release strong reference to character so it can be GC'd after respawn.
    -- FilterDescendantsInstances is rebuilt lazily on next startUndergroundRecovery call.
    _undergroundFilterChar = nil
end

local function syncUndergroundRecovery()
    if isAnySpeedFeatureActive() then startUndergroundRecovery() else stopUndergroundRecovery() end
end

local playerCharacters = {}

local function updatePlayerCharacterCache(player)
    if player.Character then playerCharacters[player.Character] = player end
end

local function onPlayerAdded(player)
    if player == localPlayer then return end
    updatePlayerCharacterCache(player)
    local c1 = player.CharacterAdded:Connect(function(char)
        playerCharacters[char] = player
    end)
    local c2 = player.CharacterRemoving:Connect(function(char)
        playerCharacters[char] = nil
    end)
    _playerCharConns[player] = { c1, c2 }
    table.insert(globalConns, c1)
    table.insert(globalConns, c2)
end

for _, player in ipairs(Players:GetPlayers()) do
    if player ~= localPlayer then onPlayerAdded(player) end
end

table.insert(globalConns, Players.PlayerAdded:Connect(onPlayerAdded))

table.insert(globalConns, Players.PlayerRemoving:Connect(function(player)
    local conns = _playerCharConns[player]
    if not conns then return end
    for _, c in ipairs(conns) do c:Disconnect() end
    _playerCharConns[player] = nil
    for i = #globalConns, 1, -1 do
        if globalConns[i] and not globalConns[i].Connected then
            table.remove(globalConns, i)
        end
    end
end))

local function isPlayerCharacterOrPart(obj)
    if playerCharacters[obj] then return true end
    local ancestor = obj.Parent
    while ancestor and ancestor ~= Workspace do
        if playerCharacters[ancestor] then return true end
        ancestor = ancestor.Parent
    end
    return false
end

local function cleanItemName(name)
    return (name:gsub("[%. _][0-9]+$", ""))
end

-- GLOBAL DISTANCE UPDATER
-- =====================================================================
local globalDistanceHeartbeat = nil
local DIST_UPDATE_INTERVAL    = 0.2

local function startGlobalDistanceUpdater()
    if globalDistanceHeartbeat then return end
    local lastUpdate = 0
    globalDistanceHeartbeat = RunService.Heartbeat:Connect(function()
        local now = tick()
        if now - lastUpdate < DIST_UPDATE_INTERVAL then return end
        lastUpdate = now
        local char = localPlayer.Character
        local root = char and char:FindFirstChild("HumanoidRootPart")
        for _, data in pairs(espInstances) do
            local pp = data.primaryPart
            if pp and pp.Parent then
                if root then
                    local dist    = (root.Position - pp.Position).Magnitude
                    local distInt = math.floor(dist + 0.5)
                    if data.lastDistInt ~= distInt then
                        data.lastDistInt  = distInt
                        data.label.Text   = data.name .. "\n" .. distInt .. " studs"
                    end
                else
                    if data.lastDistInt ~= -1 then
                        data.lastDistInt = -1
                        data.label.Text  = data.name .. "\n?? studs"
                    end
                end
            end
        end
    end)
end

local function stopGlobalDistanceUpdater()
    if globalDistanceHeartbeat then
        globalDistanceHeartbeat:Disconnect()
        globalDistanceHeartbeat = nil
    end
end

local TEAM_VALUE_TYPES = { "StringValue", "IntValue", "NumberValue" }
local function isTeamValue(obj)
    for _, t in ipairs(TEAM_VALUE_TYPES) do
        if obj:IsA(t) then
            local lower = obj.Name:lower()
            if lower:match("team") or lower == "faction" or lower == "side"
                or lower == "alliance" or lower == "group" then
                return true
            end
        end
    end
    return false
end

local function getTeamValueOf(plr)
    for _, obj in ipairs(plr:GetChildren()) do
        if isTeamValue(obj) then return tostring(obj.Value) end
    end
    if plr.Character then
        for _, obj in ipairs(plr.Character:GetChildren()) do
            if isTeamValue(obj) then return tostring(obj.Value) end
        end
        for _, obj in ipairs(plr.Character:GetDescendants()) do
            if isTeamValue(obj) then return tostring(obj.Value) end
        end
    end
    return nil
end

local function playerHasTeam(player)
    if player.TeamColor ~= BrickColor.new("White") then return true end
    if player.Team then return true end
    return getTeamValueOf(player) ~= nil
end

local function isSameTeam(playerA, playerB)
    local whiteBC = BrickColor.new("White")
    if playerA.TeamColor ~= whiteBC and playerB.TeamColor ~= whiteBC then
        return playerA.TeamColor == playerB.TeamColor
    end
    if playerA.Team and playerB.Team then
        return playerA.Team == playerB.Team
    end
    if playerA.Team and playerB.TeamColor ~= whiteBC then
        return playerA.Team.TeamColor == playerB.TeamColor
    end
    if playerB.Team and playerA.TeamColor ~= whiteBC then
        return playerB.Team.TeamColor == playerA.TeamColor
    end
    local valA, valB = getTeamValueOf(playerA), getTeamValueOf(playerB)
    if valA ~= nil and valB ~= nil then return valA == valB end
    return false
end

local function getPlayerESPColor(player)
    if not playerHasTeam(localPlayer) then return Color3.new(1, 1, 1) end
    if isSameTeam(localPlayer, player) then
        return Color3.new(0, 1, 0)
    end
    return Color3.new(1, 0, 0)
end

local function destroyESPInstance(inst)
    local data = espInstances[inst]
    if not data then return end
    if data.highlight and data.highlight.Parent then data.highlight:Destroy() end
    if data.billboard and data.billboard.Parent then data.billboard:Destroy() end
    if data.connections then
        for _, conn in ipairs(data.connections) do conn:Disconnect() end
    end
    data.instance    = nil
    data.primaryPart = nil
    data.highlight   = nil  
    data.billboard   = nil  
    data.label       = nil
    data.extraData   = nil
    data.connections = {}
    espInstances[inst] = nil
end

local function createESP(instance, espType, color, name, extraData)
    if espInstances[instance] then return espInstances[instance].type == espType end

    local primaryPart
    if instance:IsA("BasePart") then
        primaryPart = instance
    elseif instance:IsA("Model") then
        primaryPart = instance.PrimaryPart or instance:FindFirstChildWhichIsA("BasePart", true)
    end
    if not primaryPart then return false end

    local displayName = (espType == ESP_TYPE.PLAYER) and name or cleanItemName(name)

    local highlight = Instance.new("Highlight")
    highlight.Adornee             = instance
    highlight.FillColor           = color
    highlight.OutlineColor        = color
    highlight.FillTransparency    = 0.5
    highlight.OutlineTransparency = 0
    highlight.DepthMode           = Enum.HighlightDepthMode.AlwaysOnTop
    highlight.Parent              = instance

    local billboard = Instance.new("BillboardGui")
    billboard.Adornee     = primaryPart
    billboard.Size        = UDim2.new(0, 200, 0, 50)
    billboard.StudsOffset = Vector3.new(0, 3, 0)
    billboard.AlwaysOnTop = true
    billboard.MaxDistance = 9e9
    billboard.Parent      = instance

    local textLabel = Instance.new("TextLabel")
    textLabel.Size                   = UDim2.new(1, 0, 1, 0)
    textLabel.BackgroundTransparency = 1
    textLabel.TextColor3             = color
    textLabel.TextStrokeColor3       = Color3.new(0, 0, 0)
    textLabel.TextStrokeTransparency = 0.3
    textLabel.Font                   = Enum.Font.SourceSansBold
    textLabel.TextSize               = 16
    textLabel.Text                   = displayName .. "\n0 studs"
    textLabel.Parent                 = billboard

    local connections = {}

    local ancestryConn
    ancestryConn = instance.AncestryChanged:Connect(function()
        if not instance:IsDescendantOf(game) then
            ancestryConn:Disconnect()
            destroyESPInstance(instance)
        end
    end)
    table.insert(connections, ancestryConn)

    if espType == ESP_TYPE.ITEM then
        local parentConn = instance:GetPropertyChangedSignal("Parent"):Connect(function()
            if instance:FindFirstAncestorOfClass("Player") then
                destroyESPInstance(instance)
            end
        end)
        table.insert(connections, parentConn)
    end

    espInstances[instance] = {
        instance    = instance,
        highlight   = highlight,
        billboard   = billboard,
        label       = textLabel,
        type        = espType,
        name        = displayName,
        primaryPart = primaryPart,
        color       = color,
        extraData   = extraData,
        connections = connections,
        lastDistInt = nil,  
    }
    return true
end

local function updatePlayerESPColor(player, newColor)
    for _, data in pairs(espInstances) do
        if data.type == ESP_TYPE.PLAYER and data.extraData and data.extraData.player == player then
            if data.color ~= newColor then
                data.color                  = newColor
                data.highlight.FillColor    = newColor
                data.highlight.OutlineColor = newColor
                data.label.TextColor3       = newColor
            end
            break
        end
    end
end

local function refreshAllPlayerColors()
    if not toggles.playersESP then return end
    for _, data in pairs(espInstances) do
        if data.type == ESP_TYPE.PLAYER and data.extraData and data.extraData.player then
            local newColor = getPlayerESPColor(data.extraData.player)
            if newColor ~= data.color then
                data.color                  = newColor
                data.highlight.FillColor    = newColor
                data.highlight.OutlineColor = newColor
                data.label.TextColor3       = newColor
            end
        end
    end
end

local _refreshColorsPending = false
local function deferRefreshAllPlayerColors()
    if _refreshColorsPending then return end
    _refreshColorsPending = true
    task.defer(function()
        _refreshColorsPending = false
        if toggles.playersESP then refreshAllPlayerColors() end
    end)
end

local function disconnectESPListeners(espType)
    if espTypeConnections[espType] then
        for _, conn in ipairs(espTypeConnections[espType]) do conn:Disconnect() end
        espTypeConnections[espType] = {}
    end
end

    if anyActive then startGlobalDistanceUpdater() else stopGlobalDistanceUpdater() end
end

local function attemptCreatePlayerESP(char, player, trackData)
    if not char or not player then return end
    if espInstances[char] then
        if trackData.waitingConn then
            trackData.waitingConn:Disconnect()
            trackData.waitingConn     = nil
            trackData.waitingForParts = false
        end
        return
    end
    local color   = getPlayerESPColor(player)
    local created = createESP(char, ESP_TYPE.PLAYER, color, player.Name, {player = player})
    if created then
        if trackData.waitingConn then
            trackData.waitingConn:Disconnect()
            trackData.waitingConn     = nil
            trackData.waitingForParts = false
        end
    else
        if not trackData.waitingForParts then
            trackData.waitingForParts = true
            trackData.waitingConn = char.DescendantAdded:Connect(function(desc)
                if desc:IsA("BasePart") and not espInstances[char] then
                    attemptCreatePlayerESP(char, player, trackData)
                end
            end)
            if char:FindFirstChildWhichIsA("BasePart", true) then
                attemptCreatePlayerESP(char, player, trackData)
            end
        end
    end
end

local function trackPlayer(player)
    if player == localPlayer then return end
    if playerTrackers[player] then return end

    local trackData = { player=player, connections={}, charConnections={}, waitingConn=nil, waitingForParts=false }

    local function handleCharacter(char)
        if char then attemptCreatePlayerESP(char, player, trackData) end
    end
    local function handleCharacterRemoving(char)
        if char and espInstances[char] then destroyESPInstance(char) end
        if trackData.waitingConn then
            trackData.waitingConn:Disconnect()
            trackData.waitingConn     = nil
            trackData.waitingForParts = false
        end
        for _, conn in ipairs(trackData.charConnections) do conn:Disconnect() end
        trackData.charConnections = {}
    end

    if player.Character then handleCharacter(player.Character) end
    table.insert(trackData.connections, player.CharacterRemoving:Connect(handleCharacterRemoving))
    table.insert(trackData.connections,
        player:GetPropertyChangedSignal("TeamColor"):Connect(function()
            if toggles.playersESP then updatePlayerESPColor(player, getPlayerESPColor(player)) end
        end))
    table.insert(trackData.connections,
        player:GetPropertyChangedSignal("Team"):Connect(function()
            if toggles.playersESP then updatePlayerESPColor(player, getPlayerESPColor(player)) end
        end))
    table.insert(trackData.connections,
        player.DescendantAdded:Connect(function()
            if toggles.playersESP then deferRefreshAllPlayerColors() end
        end))

    local function watchCharTV(char)
        if not char then return end
        table.insert(trackData.charConnections,
            char.DescendantAdded:Connect(function()
                if toggles.playersESP then deferRefreshAllPlayerColors() end
            end))
    end
    if player.Character then watchCharTV(player.Character) end

    table.insert(trackData.connections,
        player.CharacterAdded:Connect(function(char)
            handleCharacter(char)
            watchCharTV(char)
            task.defer(function()
                if toggles.playersESP and playerTrackers[player] then
                    updatePlayerESPColor(player, getPlayerESPColor(player))
                end
            end)
        end))

    playerTrackers[player] = trackData
end

local function untrackPlayer(player)
    local trackData = playerTrackers[player]
    if not trackData then return end
    if trackData.waitingConn then trackData.waitingConn:Disconnect() end
    for _, conn in ipairs(trackData.connections) do conn:Disconnect() end
    for _, conn in ipairs(trackData.charConnections) do conn:Disconnect() end
    local toDestroy = {}
    for inst, data in pairs(espInstances) do
        if data.type == ESP_TYPE.PLAYER and data.extraData and data.extraData.player == player then
            toDestroy[#toDestroy+1] = inst
        end
    end
    for _, inst in ipairs(toDestroy) do destroyESPInstance(inst) end
    playerTrackers[player] = nil
end

function togglePlayersESP(enable)
    if enable then
        for _, player in ipairs(Players:GetPlayers()) do
            if player ~= localPlayer then trackPlayer(player) end
        end
        table.insert(espTypeConnections.players, Players.PlayerAdded:Connect(function(player)
            if player ~= localPlayer then trackPlayer(player) end
        end))
        table.insert(espTypeConnections.players, Players.PlayerRemoving:Connect(untrackPlayer))
        table.insert(espTypeConnections.players,
            localPlayer:GetPropertyChangedSignal("TeamColor"):Connect(refreshAllPlayerColors))
        table.insert(espTypeConnections.players,
            localPlayer:GetPropertyChangedSignal("Team"):Connect(refreshAllPlayerColors))
        table.insert(espTypeConnections.players,
            localPlayer.DescendantAdded:Connect(function()
                if toggles.playersESP then deferRefreshAllPlayerColors() end
            end))

        local function watchLocalChar(char)
            for _, conn in ipairs(localCharESPConns) do conn:Disconnect() end
            localCharESPConns = {}
            if not char then return end
            table.insert(localCharESPConns,
                char.DescendantAdded:Connect(function()
                    if toggles.playersESP then deferRefreshAllPlayerColors() end
                end))
        end
        if localPlayer.Character then watchLocalChar(localPlayer.Character) end
        table.insert(espTypeConnections.players,
            localPlayer.CharacterAdded:Connect(function(char)
                watchLocalChar(char)
                refreshAllESP()
            end))
    else
        local toUntrack = {}
        for player in pairs(playerTrackers) do toUntrack[#toUntrack+1] = player end
        for _, player in ipairs(toUntrack) do untrackPlayer(player) end
        disconnectESPListeners("players")
        for _, conn in ipairs(localCharESPConns) do conn:Disconnect() end
        localCharESPConns = {}
    end
    syncGlobalUpdater()
end

local function isExcludedFromItemESP(obj)
    if isPlayerCharacterOrPart(obj) then return true end
    if obj:FindFirstChildWhichIsA("Humanoid", true) then return true end
    local ancestor = obj.Parent
    while ancestor and ancestor ~= Workspace do
        if ancestor:IsA("Model") and ancestor:FindFirstChildWhichIsA("Humanoid", true) then
            return true
        end
        ancestor = ancestor.Parent
    end
    return false
end

local function getRootItemContainer(obj)
    local best    = nil
    local current = obj.Parent
    while current and current ~= Workspace do
        if current:FindFirstAncestorOfClass("Player") then return nil end
        if current:IsA("Model") then
            if current:FindFirstChildWhichIsA("Humanoid", true) then break end
            best = current
        elseif current:IsA("BasePart") and not best then
            best = current
        end
        current = current.Parent
    end
    return best
end

local function findInteractionComponent(obj)
    for _, child in ipairs(obj:GetChildren()) do
        if child:IsA("ProximityPrompt") or child:IsA("ClickDetector") then return child end
    end
    local pp = obj:FindFirstChildWhichIsA("ProximityPrompt", true)
    if pp then return pp end
    return obj:FindFirstChildWhichIsA("ClickDetector", true)
end

local function classifyAsItem(obj)
    if isExcludedFromItemESP(obj) then return false end
    if obj:IsA("Tool") then
        return (not obj:FindFirstAncestorOfClass("Player")
            and not isPlayerCharacterOrPart(obj)) and obj or false
    end
    if obj:IsA("ProximityPrompt") or obj:IsA("ClickDetector") then
        local root = getRootItemContainer(obj)
        if root and not isExcludedFromItemESP(root) then return root end
        local parent = obj.Parent
        if parent and (parent:IsA("Model") or parent:IsA("BasePart"))
            and not isExcludedFromItemESP(parent) then return parent end
        return false
    end
    if obj:IsA("Model") or obj:IsA("BasePart") then
        if findInteractionComponent(obj) then return obj end
    end
    return false
end

function toggleItemsESP(enable)
    if enable then
        local function flushDeadItemESP()
            local toDestroy = {}
            for inst, data in pairs(espInstances) do
                if data.type == ESP_TYPE.ITEM and not inst:IsDescendantOf(game) then
                    toDestroy[#toDestroy+1] = inst
                end
            end
            for _, inst in ipairs(toDestroy) do destroyESPInstance(inst) end
        end
        local function scanItems()
            for _, obj in ipairs(Workspace:GetDescendants()) do
                local container = classifyAsItem(obj)
                if container and not espInstances[container] then
                    createESP(container, ESP_TYPE.ITEM, Color3.new(1,1,0), container.Name)
                end
            end
        end
        scanItems()
        local itemPendingQueue = {}
        local itemDrainQueued  = false
        local function itemDrainQueue()
            itemDrainQueued = false
            if not toggles.itemsESP then itemPendingQueue = {}; return end
            for i = 1, #itemPendingQueue do
                local obj = itemPendingQueue[i]
                if obj and obj.Parent then
                    local container = classifyAsItem(obj)
                    if container and not espInstances[container] then
                        createESP(container, ESP_TYPE.ITEM, Color3.new(1,1,0), container.Name)
                    end
                end
            end
            itemPendingQueue = {}
        end
        table.insert(espTypeConnections.items, Workspace.DescendantAdded:Connect(function(obj)
            if not toggles.itemsESP then return end
            local cls = obj.ClassName
            if cls ~= "Tool" and cls ~= "ProximityPrompt" and cls ~= "ClickDetector"
            and cls ~= "Model" and not obj:IsA("BasePart") then return end
            itemPendingQueue[#itemPendingQueue + 1] = obj
            if not itemDrainQueued then
                itemDrainQueued = true
                task.defer(itemDrainQueue)
            end
        end))
        table.insert(espTypeConnections.items, localPlayer.CharacterAdded:Connect(function()
            if toggles.itemsESP then flushDeadItemESP(); scanItems() end
        end))
    else
        local toDestroy = {}
        for inst, data in pairs(espInstances) do
            if data.type == ESP_TYPE.ITEM then toDestroy[#toDestroy+1] = inst end
        end
        for _, inst in ipairs(toDestroy) do destroyESPInstance(inst) end
        disconnectESPListeners("items")
    end
    syncGlobalUpdater()
end

local function isNPC(model)
    if not model:IsA("Model") then return false end
    if not model:FindFirstChildWhichIsA("Humanoid", true) then return false end
    return not isPlayerCharacterOrPart(model)
end

function toggleNPCsESP(enable)
    if enable then
        local npcPlayerConns = {}

        local function tryCreateNPCESP(model)
            if not model or espInstances[model] then return end
            if not isNPC(model) then return end
            createESP(model, ESP_TYPE.NPC, Color3.new(0,1,1), model.Name)
        end
        local function flushDeadNPCESP()
            local toDestroy = {}
            for inst, data in pairs(espInstances) do
                if data.type == ESP_TYPE.NPC and not inst:IsDescendantOf(game) then
                    toDestroy[#toDestroy+1] = inst
                end
            end
            for _, inst in ipairs(toDestroy) do destroyESPInstance(inst) end
        end
        local function scanNPCs()
            for _, obj in ipairs(Workspace:GetDescendants()) do tryCreateNPCESP(obj) end
        end
        local function onPlayerCharacterAdded(char)
            if espInstances[char] and espInstances[char].type == ESP_TYPE.NPC then
                destroyESPInstance(char)
            end
        end
        local function registerPlayerNPCConn(player)
            if npcPlayerConns[player] then return end
            local conn = player.CharacterAdded:Connect(onPlayerCharacterAdded)
            npcPlayerConns[player] = conn
            table.insert(espTypeConnections.npcs, conn)
        end

        scanNPCs()
        local npcPendingQueue = {}
        local npcDrainQueued  = false
        local function npcDrainQueue()
            npcDrainQueued = false
            if not toggles.npcESP then npcPendingQueue = {}; return end
            for i = 1, #npcPendingQueue do
                local obj = npcPendingQueue[i]
                if obj and obj.Parent then
                    if isNPC(obj) and not espInstances[obj] then
                        tryCreateNPCESP(obj)
                    elseif obj:IsA("Humanoid") then
                        local npcModel = obj:FindFirstAncestorWhichIsA("Model")
                        if npcModel and isNPC(npcModel) and not espInstances[npcModel] then
                            tryCreateNPCESP(npcModel)
                        end
                    end
                end
            end
            npcPendingQueue = {}
        end
        table.insert(espTypeConnections.npcs, Workspace.DescendantAdded:Connect(function(obj)
            if not toggles.npcESP then return end
            local cls = obj.ClassName
            if cls ~= "Model" and cls ~= "Humanoid" then return end
            npcPendingQueue[#npcPendingQueue + 1] = obj
            if not npcDrainQueued then
                npcDrainQueued = true
                task.defer(npcDrainQueue)
            end
        end))
        table.insert(espTypeConnections.npcs, localPlayer.CharacterAdded:Connect(function()
            if toggles.npcESP then flushDeadNPCESP(); scanNPCs() end
        end))

        for _, player in ipairs(Players:GetPlayers()) do
            registerPlayerNPCConn(player)
        end
        table.insert(espTypeConnections.npcs, Players.PlayerAdded:Connect(function(player)
            registerPlayerNPCConn(player)
        end))
        table.insert(espTypeConnections.npcs, Players.PlayerRemoving:Connect(function(player)
            local conn = npcPlayerConns[player]
            if conn then
                conn:Disconnect()
                npcPlayerConns[player] = nil
                for i = #espTypeConnections.npcs, 1, -1 do
                    if not espTypeConnections.npcs[i].Connected then
                        table.remove(espTypeConnections.npcs, i)
                    end
                end
            end
        end))
    else
        local toDestroy = {}
        for inst, data in pairs(espInstances) do
            if data.type == ESP_TYPE.NPC then toDestroy[#toDestroy+1] = inst end
        end
        for _, inst in ipairs(toDestroy) do destroyESPInstance(inst) end
        disconnectESPListeners("npcs")
    end
    syncGlobalUpdater()
end

local HELD_ITEM_COLOR = Color3.new(1, 0.5, 0)

local function tryCreateHeldItemESP(tool, player)
    if not tool:IsA("Tool") then return end
    if espInstances[tool] and espInstances[tool].type ~= ESP_TYPE.HELD_ITEM then return end
    createESP(tool, ESP_TYPE.HELD_ITEM, HELD_ITEM_COLOR, tool.Name, {player = player, tool = tool})
end

local function trackPlayerHeldItems(player)
    if heldItemTrackers[player] then return end
    local tracker = { player=player, connections={}, charConnections={} }

    local function onCharacterAdded(char)
        for _, conn in ipairs(tracker.charConnections) do conn:Disconnect() end
        tracker.charConnections = {}

        for _, child in ipairs(char:GetChildren()) do
            if child:IsA("Tool") then tryCreateHeldItemESP(child, player) end
        end
        table.insert(tracker.charConnections, char.ChildAdded:Connect(function(child)
            if child:IsA("Tool") then tryCreateHeldItemESP(child, player) end
        end))
        table.insert(tracker.charConnections, char.ChildRemoved:Connect(function(child)
            if child:IsA("Tool") and espInstances[child] then destroyESPInstance(child) end
        end))
    end

    local function onCharacterRemoving(char)
        for _, conn in ipairs(tracker.charConnections) do conn:Disconnect() end
        tracker.charConnections = {}
        if char then
            for _, child in ipairs(char:GetChildren()) do
                if child:IsA("Tool") and espInstances[child] then destroyESPInstance(child) end
            end
        end
    end

    if player.Character then onCharacterAdded(player.Character) end
    table.insert(tracker.connections, player.CharacterAdded:Connect(onCharacterAdded))
    table.insert(tracker.connections, player.CharacterRemoving:Connect(onCharacterRemoving))
    heldItemTrackers[player] = tracker
end

local function untrackPlayerHeldItems(player)
    local tracker = heldItemTrackers[player]
    if not tracker then return end
    for _, conn in ipairs(tracker.connections) do conn:Disconnect() end
    for _, conn in ipairs(tracker.charConnections) do conn:Disconnect() end
    local toDestroy = {}
    for inst, data in pairs(espInstances) do
        if data.type == ESP_TYPE.HELD_ITEM and data.extraData and data.extraData.player == player then
            toDestroy[#toDestroy+1] = inst
        end
    end
    for _, inst in ipairs(toDestroy) do destroyESPInstance(inst) end
    heldItemTrackers[player] = nil
end

function toggleHeldItemsESP(enable)
    if enable then
        for _, player in ipairs(Players:GetPlayers()) do trackPlayerHeldItems(player) end
        table.insert(espTypeConnections.helditems, Players.PlayerAdded:Connect(trackPlayerHeldItems))
        table.insert(espTypeConnections.helditems, Players.PlayerRemoving:Connect(untrackPlayerHeldItems))
    else
        local toUntrack = {}
        for player in pairs(heldItemTrackers) do toUntrack[#toUntrack+1] = player end
        for _, player in ipairs(toUntrack) do untrackPlayerHeldItems(player) end
        disconnectESPListeners("helditems")
    end
    syncGlobalUpdater()
end

function toggleNoclip(enable)
    if enable then
        local function disablePartCollision(part)
            if part:IsA("BasePart") then
                if noclipOriginalStates[part] == nil then
                    noclipOriginalStates[part] = part.CanCollide
                end
                part.CanCollide = false
                noclipPartsCache[part] = true
                local partAncConn
                partAncConn = part.AncestryChanged:Connect(function()
                    if part:IsDescendantOf(game) then return end
                    noclipPartsCache[part]     = nil
                    noclipOriginalStates[part] = nil
                    partAncConn:Disconnect()
                end)
                table.insert(noclipPartConns, partAncConn)
            end
        end

        local function applyNoclipToChar(char)
            if noclipCharDescConn then
                noclipCharDescConn:Disconnect()
                noclipCharDescConn = nil
            end
            for _, c in ipairs(noclipPartConns) do c:Disconnect() end
            noclipPartConns      = {}
            noclipPartsCache     = {}
            noclipOriginalStates = {}
            for _, part in ipairs(char:GetDescendants()) do disablePartCollision(part) end
            noclipCharDescConn = char.DescendantAdded:Connect(disablePartCollision)
        end

        if localPlayer.Character then applyNoclipToChar(localPlayer.Character) end
        table.insert(noclipConnections, localPlayer.CharacterAdded:Connect(function(newChar)
            if toggles.noclip then applyNoclipToChar(newChar) end
        end))

        noclipSteppedConn = RunService.Stepped:Connect(function()
            if not toggles.noclip then return end
            for part in pairs(noclipPartsCache) do
                if part.CanCollide then
                    part.CanCollide = false
                end
            end
        end)
    else
        if noclipSteppedConn then noclipSteppedConn:Disconnect(); noclipSteppedConn = nil end
        if noclipCharDescConn then noclipCharDescConn:Disconnect(); noclipCharDescConn = nil end
        for _, conn in ipairs(noclipPartConns)  do conn:Disconnect() end; noclipPartConns  = {}
        for _, conn in ipairs(noclipConnections) do conn:Disconnect() end; noclipConnections = {}
        noclipPartsCache = {}
        for part, original in pairs(noclipOriginalStates) do
            if part and part.Parent then part.CanCollide = original end
        end
        noclipOriginalStates = {}
    end
end

function toggleTPTool(enable)
    for _, conn in ipairs(tpToolConns) do conn:Disconnect() end
    tpToolConns = {}
    for _, conn in ipairs(_tpObjConns) do conn:Disconnect() end
    _tpObjConns = {}
    if toggles.tpToolObject then
        pcall(function() toggles.tpToolObject:Destroy() end)
        toggles.tpToolObject = nil
    end

    if not enable then return end

    local function buildTool()
        for _, conn in ipairs(_tpObjConns) do conn:Disconnect() end
        _tpObjConns = {}
        if toggles.tpToolObject then
            pcall(function() toggles.tpToolObject:Destroy() end)
            toggles.tpToolObject = nil
        end

        local tool = Instance.new("Tool")
        tool.Name           = "Teleport Tool"
        tool.RequiresHandle = false
        tool.CanBeDropped   = true

        local mouse  = localPlayer:GetMouse()
        local marker = nil
        local tpRayParams = RaycastParams.new()
        tpRayParams.FilterType = Enum.RaycastFilterType.Exclude

        local c1 = tool.Equipped:Connect(function()
            if marker then marker:Destroy() end
            marker              = Instance.new("Part")
            marker.Anchored     = true
            marker.CanCollide   = false
            marker.Size         = Vector3.new(2, 0.2, 2)
            marker.BrickColor   = BrickColor.new("Bright blue")
            marker.Material     = Enum.Material.Neon
            marker.Transparency = 0.5
            marker.Parent       = workspace
        end)
        local c2 = tool.Unequipped:Connect(function()
            if marker then marker:Destroy(); marker = nil end
        end)
        local c3 = tool.Activated:Connect(function()
            if not marker then return end
            local char = localPlayer.Character
            if not char or not char:FindFirstChild("HumanoidRootPart") then return end
            if not mouse.Hit then return end

            local target    = mouse.Hit.Position
            tpRayParams.FilterDescendantsInstances = { char }
            local rayResult = workspace:Raycast(target + Vector3.new(0,10,0), Vector3.new(0,-20,0), tpRayParams)
            if rayResult then target = rayResult.Position + Vector3.new(0,3,0) end
            marker.CFrame = CFrame.new(target)
            task.wait(0.05)
            char.HumanoidRootPart.CFrame = CFrame.new(target)
        end)
        local c4 = tool.AncestryChanged:Connect(function()
            if not tool.Parent and marker then
                marker:Destroy(); marker = nil
            end
        end)
        _tpObjConns[1] = c1
        _tpObjConns[2] = c2
        _tpObjConns[3] = c3
        _tpObjConns[4] = c4

        toggles.tpToolObject = tool

        local backpack = localPlayer:FindFirstChild("Backpack")
        if backpack then
            tool.Parent = backpack
        else
            tool.Parent = localPlayer
        end
    end

    buildTool()

    local respawnConn = localPlayer.CharacterAdded:Connect(function()
        if not toggles.tpTool then return end
        task.wait(0.15)
        if not toggles.tpTool then return end
        local backpack = localPlayer:FindFirstChild("Backpack")
        local char     = localPlayer.Character
        local hasTool  = (backpack and backpack:FindFirstChild("Teleport Tool"))
                      or (char and char:FindFirstChild("Teleport Tool"))
        if not hasTool then
            buildTool()
        end
    end)
    tpToolConns[1] = respawnConn
end

local function isPartOfAnyCharacter(obj)
    local current = obj
    while current and current ~= Workspace do
        if playerCharacters[current] then return true end
        current = current.Parent
    end
    return false
end

local function applyFPSBoostToInstance(inst)
    if isPartOfAnyCharacter(inst) then return end

    if inst:IsA("Part") or inst:IsA("Union") or inst:IsA("CornerWedgePart")
        or inst:IsA("TrussPart") or inst:IsA("MeshPart") then
        if not fpsBoostData.partProps[inst] then
            fpsBoostData.partProps[inst] = {
                Material    = inst.Material,
                Reflectance = inst.Reflectance,
                TextureID   = inst:IsA("MeshPart") and inst.TextureID or nil,
            }
        end
        inst.Material    = Enum.Material.Plastic
        inst.Reflectance = 0
        if inst:IsA("MeshPart") then inst.TextureID = "" end
    end
    if inst:IsA("Decal") or inst:IsA("Texture") then
        if not fpsBoostData.decalProps[inst] then
            fpsBoostData.decalProps[inst] = { Transparency = inst.Transparency }
        end
        inst.Transparency = 1
    end
    if inst:IsA("ParticleEmitter") or inst:IsA("Trail") then
        if not fpsBoostData.particleProps[inst] then
            fpsBoostData.particleProps[inst] = { Enabled = inst.Enabled }
        end
        inst.Enabled = false
    end
    if inst:IsA("Fire") or inst:IsA("SpotLight") or inst:IsA("Smoke") or inst:IsA("Sparkles") then
        if not fpsBoostData.lightProps[inst] then
            fpsBoostData.lightProps[inst] = { Enabled = inst.Enabled }
        end
        inst.Enabled = false
    end
    if inst:IsA("Explosion") then
        if not fpsBoostData.effectProps[inst] then
            fpsBoostData.effectProps[inst] = {
                BlastPressure = inst.BlastPressure,
                BlastRadius   = inst.BlastRadius,
            }
        end
        inst.BlastPressure = 1
        inst.BlastRadius   = 1
    end
    if inst:IsA("BlurEffect") or inst:IsA("SunRaysEffect") or inst:IsA("ColorCorrectionEffect")
        or inst:IsA("BloomEffect") or inst:IsA("DepthOfFieldEffect") then
        if not fpsBoostData.effectProps[inst] then
            fpsBoostData.effectProps[inst] = { Enabled = inst.Enabled }
        end
        inst.Enabled = false
    end
end

local function restoreFPSBoostData()
    for inst, props in pairs(fpsBoostData.partProps) do
        if inst and inst.Parent then
            inst.Material    = props.Material
            inst.Reflectance = props.Reflectance
            if props.TextureID ~= nil then inst.TextureID = props.TextureID end
        end
    end
    fpsBoostData.partProps = {}
    for inst, props in pairs(fpsBoostData.decalProps) do
        if inst and inst.Parent then inst.Transparency = props.Transparency end
    end
    fpsBoostData.decalProps = {}
    for inst, props in pairs(fpsBoostData.effectProps) do
        if inst and inst.Parent then
            if props.Enabled ~= nil then
                inst.Enabled = props.Enabled
            end
            if props.BlastPressure ~= nil then
                pcall(function() inst.BlastPressure = props.BlastPressure end)
            end
            if props.BlastRadius ~= nil then
                pcall(function() inst.BlastRadius = props.BlastRadius end)
            end
        end
    end
    fpsBoostData.effectProps = {}
    for inst, props in pairs(fpsBoostData.lightProps) do
        if inst and inst.Parent then inst.Enabled = props.Enabled end
    end
    fpsBoostData.lightProps = {}
    for inst, props in pairs(fpsBoostData.particleProps) do
        if inst and inst.Parent then inst.Enabled = props.Enabled end
    end
    fpsBoostData.particleProps = {}
end

function toggleFPSBoost(enable)
    if enable then
        captureOriginalLighting()

        if not fpsBoostProperties.Terrain then
            fpsBoostProperties.Terrain = {
                WaterWaveSize     = Workspace.Terrain.WaterWaveSize,
                WaterWaveSpeed    = Workspace.Terrain.WaterWaveSpeed,
                WaterReflectance  = Workspace.Terrain.WaterReflectance,
                WaterTransparency = Workspace.Terrain.WaterTransparency,
            }
        end
        pcall(function() Workspace.Terrain.WaterWaveSize     = 0 end)
        pcall(function() Workspace.Terrain.WaterWaveSpeed    = 0 end)
        pcall(function() Workspace.Terrain.WaterReflectance  = 0 end)
        pcall(function() Workspace.Terrain.WaterTransparency = 0 end)

        Lighting.GlobalShadows = false
        Lighting.FogEnd        = 100000
        Lighting.Brightness    = 0

        if not fpsBoostProperties.QualityLevel then
            pcall(function()
                fpsBoostProperties.QualityLevel = settings().Rendering.QualityLevel
            end)
        end
        pcall(function()
            settings().Rendering.QualityLevel = Enum.QualityLevel.Level01
        end)

        for _, obj in ipairs(Workspace:GetDescendants()) do applyFPSBoostToInstance(obj) end
        for _, obj in ipairs(Lighting:GetDescendants())  do applyFPSBoostToInstance(obj) end

        local fpsPendingQueue  = {}
        local fpsDrainQueued   = false
        local function fpsDrainQueue()
            fpsDrainQueued = false
            if not toggles.fpsBoost then fpsPendingQueue = {}; return end
            for i = 1, #fpsPendingQueue do
                applyFPSBoostToInstance(fpsPendingQueue[i])
            end
            fpsPendingQueue = {}
        end
        if fpsBoostData.descendantConn then
            fpsBoostData.descendantConn:Disconnect()
            fpsBoostData.descendantConn = nil
        end
        fpsBoostData.descendantConn = Workspace.DescendantAdded:Connect(function(obj)
            if not toggles.fpsBoost then return end
            local cls = obj.ClassName
            if cls ~= "Part"          and cls ~= "Union"          and cls ~= "CornerWedgePart"
            and cls ~= "TrussPart"    and cls ~= "MeshPart"
            and cls ~= "Decal"        and cls ~= "Texture"
            and cls ~= "ParticleEmitter" and cls ~= "Trail"       and cls ~= "Beam"
            and cls ~= "Fire"         and cls ~= "Smoke"          and cls ~= "Sparkles"
            and cls ~= "PointLight"   and cls ~= "SurfaceLight"   and cls ~= "SpotLight" then
                return
            end
            fpsPendingQueue[#fpsPendingQueue + 1] = obj
            if not fpsDrainQueued then
                fpsDrainQueued = true
                task.defer(fpsDrainQueue)
            end
        end)

        if not fpsBoostData.removingConn then
            fpsBoostData.removingConn = Workspace.DescendantRemoving:Connect(function(obj)
                if fpsBoostData.partProps[obj]     then fpsBoostData.partProps[obj]     = nil end
                if fpsBoostData.decalProps[obj]    then fpsBoostData.decalProps[obj]    = nil end
                if fpsBoostData.effectProps[obj]   then fpsBoostData.effectProps[obj]   = nil end
                if fpsBoostData.lightProps[obj]    then fpsBoostData.lightProps[obj]    = nil end
                if fpsBoostData.particleProps[obj] then fpsBoostData.particleProps[obj] = nil end
            end)
        end
        if not fpsBoostData.mapRemovedConn then
            fpsBoostData.mapRemovedConn = Workspace.ChildRemoved:Connect(function(removed)
                if not (removed:IsA("Model") or removed:IsA("Folder")) then return end
                for inst in pairs(fpsBoostData.partProps)     do if inst:IsDescendantOf(removed) then fpsBoostData.partProps[inst]     = nil end end
                for inst in pairs(fpsBoostData.decalProps)    do if inst:IsDescendantOf(removed) then fpsBoostData.decalProps[inst]    = nil end end
                for inst in pairs(fpsBoostData.effectProps)   do if inst:IsDescendantOf(removed) then fpsBoostData.effectProps[inst]   = nil end end
                for inst in pairs(fpsBoostData.lightProps)    do if inst:IsDescendantOf(removed) then fpsBoostData.lightProps[inst]    = nil end end
                for inst in pairs(fpsBoostData.particleProps) do if inst:IsDescendantOf(removed) then fpsBoostData.particleProps[inst] = nil end end
            end)
        end

        if not fpsBoostData.charAddedConn then
            fpsBoostData.charAddedConn = localPlayer.CharacterAdded:Connect(function()
                if not toggles.fpsBoost then return end
                restoreFPSBoostData()
                for _, obj in ipairs(Workspace:GetDescendants()) do applyFPSBoostToInstance(obj) end
                for _, obj in ipairs(Lighting:GetDescendants())  do applyFPSBoostToInstance(obj) end
                Lighting.GlobalShadows = false
                Lighting.FogEnd        = 100000
                Lighting.Brightness    = 0
                pcall(function()
                    settings().Rendering.QualityLevel = Enum.QualityLevel.Level01
                end)
            end)
        end
    else
        if fpsBoostProperties.Terrain then
            pcall(function() Workspace.Terrain.WaterWaveSize     = fpsBoostProperties.Terrain.WaterWaveSize     end)
            pcall(function() Workspace.Terrain.WaterWaveSpeed    = fpsBoostProperties.Terrain.WaterWaveSpeed    end)
            pcall(function() Workspace.Terrain.WaterReflectance  = fpsBoostProperties.Terrain.WaterReflectance  end)
            pcall(function() Workspace.Terrain.WaterTransparency = fpsBoostProperties.Terrain.WaterTransparency end)
            fpsBoostProperties.Terrain = nil
        end
        if fpsBoostProperties.QualityLevel ~= nil then
            pcall(function()
                settings().Rendering.QualityLevel = fpsBoostProperties.QualityLevel
            end)
            fpsBoostProperties.QualityLevel = nil
        end
        if not toggles.fullbright and originalLightingProperties then
            Lighting.GlobalShadows = originalLightingProperties.GlobalShadows
            Lighting.FogEnd        = originalLightingProperties.FogEnd
            pcall(function() Lighting.Brightness = originalLightingProperties.Brightness end)
        end
        if fpsBoostData.descendantConn  then fpsBoostData.descendantConn:Disconnect();  fpsBoostData.descendantConn  = nil end
        if fpsBoostData.charAddedConn   then fpsBoostData.charAddedConn:Disconnect();   fpsBoostData.charAddedConn   = nil end
        if fpsBoostData.removingConn    then fpsBoostData.removingConn:Disconnect();    fpsBoostData.removingConn    = nil end
        if fpsBoostData.mapRemovedConn  then fpsBoostData.mapRemovedConn:Disconnect();  fpsBoostData.mapRemovedConn  = nil end
        restoreFPSBoostData()
        releaseOriginalLighting()
    end
end

function toggleWalkspeed(enable)
    if enable then
        local function applyWalkspeed(humanoid)
            if not humanoid then return end
            if not walkspeedOriginalValues[humanoid] then
                walkspeedOriginalValues[humanoid] = humanoid.WalkSpeed
            end
            humanoid.WalkSpeed = WALK_SPEED

            if _wsHumPropConn then _wsHumPropConn:Disconnect() end
            if _wsHumAncConn  then _wsHumAncConn:Disconnect()  end

            _wsHumPropConn = humanoid:GetPropertyChangedSignal("WalkSpeed"):Connect(function()
                if not toggles.walkspeed then _wsHumPropConn:Disconnect(); _wsHumPropConn = nil; return end
                if humanoid.WalkSpeed ~= WALK_SPEED then
                    humanoid.WalkSpeed = WALK_SPEED
                end
            end)

            _wsHumAncConn = humanoid.AncestryChanged:Connect(function()
                if humanoid:IsDescendantOf(game) then return end
                walkspeedOriginalValues[humanoid] = nil
                _wsHumAncConn:Disconnect()
                _wsHumAncConn = nil
            end)
        end

        local function applyToChar(char)
            if not char then return end
            local humanoid = char:FindFirstChildWhichIsA("Humanoid", true)
            if humanoid then applyWalkspeed(humanoid) end
        end

        local function setupCharSlots(newChar)
            if _wsCharDescConn then _wsCharDescConn:Disconnect(); _wsCharDescConn = nil end
            if _wsCharAncConn  then _wsCharAncConn:Disconnect();  _wsCharAncConn  = nil end

            applyToChar(newChar)

            _wsCharDescConn = newChar.DescendantAdded:Connect(function(desc)
                if toggles.walkspeed and desc:IsA("Humanoid") then
                    applyWalkspeed(desc)
                end
            end)

            _wsCharAncConn = newChar.AncestryChanged:Connect(function()
                if not newChar.Parent then
                    if _wsCharDescConn then _wsCharDescConn:Disconnect(); _wsCharDescConn = nil end
                    _wsCharAncConn:Disconnect(); _wsCharAncConn = nil
                end
            end)
        end

        applyToChar(localPlayer.Character)
        if localPlayer.Character then
            local char = localPlayer.Character
            if _wsCharDescConn then _wsCharDescConn:Disconnect() end
            if _wsCharAncConn  then _wsCharAncConn:Disconnect()  end
            _wsCharDescConn = char.DescendantAdded:Connect(function(desc)
                if toggles.walkspeed and desc:IsA("Humanoid") then applyWalkspeed(desc) end
            end)
            _wsCharAncConn = char.AncestryChanged:Connect(function()
                if not char.Parent then
                    if _wsCharDescConn then _wsCharDescConn:Disconnect(); _wsCharDescConn = nil end
                    _wsCharAncConn:Disconnect(); _wsCharAncConn = nil
                end
            end)
        end

        if _wsCharAddedConn then _wsCharAddedConn:Disconnect() end
        _wsCharAddedConn = localPlayer.CharacterAdded:Connect(function(newChar)
            if not toggles.walkspeed then return end
            setupCharSlots(newChar)
        end)

        syncUndergroundRecovery()
    else
        if _wsCharAddedConn then _wsCharAddedConn:Disconnect(); _wsCharAddedConn = nil end
        if _wsHumPropConn   then _wsHumPropConn:Disconnect();   _wsHumPropConn   = nil end
        if _wsHumAncConn    then _wsHumAncConn:Disconnect();    _wsHumAncConn    = nil end
        if _wsCharDescConn  then _wsCharDescConn:Disconnect();  _wsCharDescConn  = nil end
        if _wsCharAncConn   then _wsCharAncConn:Disconnect();   _wsCharAncConn   = nil end
        for humanoid, original in pairs(walkspeedOriginalValues) do
            if humanoid and humanoid.Parent then humanoid.WalkSpeed = original end
        end
        walkspeedOriginalValues = {}
        syncUndergroundRecovery()
    end
end

local FLY_GROUND_BUFFER = 4

local _flyGroundRayParams        = RaycastParams.new()
_flyGroundRayParams.FilterType   = Enum.RaycastFilterType.Exclude
local _flyGroundFilterChar       = nil

local function flyGroundCheck(rootPart, char, moveDir)
    if moveDir.Y >= 0 then return moveDir end
    if _flyGroundFilterChar ~= char then
        _flyGroundFilterChar = char
        _flyGroundRayParams.FilterDescendantsInstances = { char }
    end
    local checkDist = FLY_GROUND_BUFFER + math.abs(moveDir.Y) * 0.2
    local result    = workspace:Raycast(rootPart.Position, Vector3.new(0, -checkDist, 0), _flyGroundRayParams)
    if result then return Vector3.new(moveDir.X, 0, moveDir.Z) end
    return moveDir
end

function toggleFly(enable)
    if enable then
        flyActive = false
        for _, conn in ipairs(flyConnections) do conn:Disconnect() end
        flyConnections = {}
        if flyLinearVelocity   then pcall(function() flyLinearVelocity:Destroy()   end); flyLinearVelocity   = nil end
        if flyAlignOrientation then pcall(function() flyAlignOrientation:Destroy() end); flyAlignOrientation = nil end
        if flyAttachment       then pcall(function() flyAttachment:Destroy()       end); flyAttachment       = nil end
        if flyAttachmentGyro   then pcall(function() flyAttachmentGyro:Destroy()   end); flyAttachmentGyro   = nil end

        local char = localPlayer.Character
        if not char then return end
        local rootPart = char:FindFirstChild("HumanoidRootPart")
        if not rootPart then return end

        flyAttachment        = Instance.new("Attachment")
        flyAttachment.Parent = rootPart

        flyLinearVelocity                        = Instance.new("LinearVelocity")
        flyLinearVelocity.Attachment0            = flyAttachment
        flyLinearVelocity.MaxForce               = math.huge
        flyLinearVelocity.VelocityConstraintMode = Enum.VelocityConstraintMode.Vector
        flyLinearVelocity.RelativeTo             = Enum.ActuatorRelativeTo.World
        flyLinearVelocity.VectorVelocity         = Vector3.new(0, 0, 0)
        flyLinearVelocity.Parent                 = rootPart

        flyAttachmentGyro        = Instance.new("Attachment")
        flyAttachmentGyro.Parent = rootPart

        flyAlignOrientation                     = Instance.new("AlignOrientation")
        flyAlignOrientation.Mode               = Enum.OrientationAlignmentMode.OneAttachment
        flyAlignOrientation.Attachment0        = flyAttachmentGyro
        flyAlignOrientation.MaxTorque          = math.huge
        flyAlignOrientation.MaxAngularVelocity = math.huge
        flyAlignOrientation.Responsiveness     = 200
        flyAlignOrientation.RigidityEnabled    = false
        flyAlignOrientation.Parent             = rootPart

        local humanoid = char:FindFirstChild("Humanoid")
        if humanoid then
            flyOrigPlatformStand   = humanoid.PlatformStand
            humanoid.PlatformStand = true
        end

        flyActive = true
        syncUndergroundRecovery()

        local moveConn
        moveConn = RunService.Heartbeat:Connect(function()
            if not flyActive or not char or not char.Parent then
                moveConn:Disconnect(); return
            end
            if not rootPart or not rootPart.Parent then return end

            if humanoid and not humanoid.PlatformStand then
                humanoid.PlatformStand = true
            end

            if not flyLinearVelocity or not flyLinearVelocity.Parent
            or not flyAlignOrientation or not flyAlignOrientation.Parent then
                if toggles.fly then
                    if flyLinearVelocity   then pcall(function() flyLinearVelocity:Destroy()   end); flyLinearVelocity   = nil end
                    if flyAlignOrientation then pcall(function() flyAlignOrientation:Destroy() end); flyAlignOrientation = nil end
                    if flyAttachment       then pcall(function() flyAttachment:Destroy()       end); flyAttachment       = nil end
                    if flyAttachmentGyro   then pcall(function() flyAttachmentGyro:Destroy()   end); flyAttachmentGyro   = nil end
                    local rp = char:FindFirstChild("HumanoidRootPart")
                    if rp then
                        flyAttachment        = Instance.new("Attachment"); flyAttachment.Parent = rp
                        flyLinearVelocity    = Instance.new("LinearVelocity")
                        flyLinearVelocity.Attachment0            = flyAttachment
                        flyLinearVelocity.MaxForce               = math.huge
                        flyLinearVelocity.VelocityConstraintMode = Enum.VelocityConstraintMode.Vector
                        flyLinearVelocity.RelativeTo             = Enum.ActuatorRelativeTo.World
                        flyLinearVelocity.VectorVelocity         = Vector3.new(0,0,0)
                        flyLinearVelocity.Parent                 = rp
                        flyAttachmentGyro    = Instance.new("Attachment"); flyAttachmentGyro.Parent = rp
                        flyAlignOrientation  = Instance.new("AlignOrientation")
                        flyAlignOrientation.Mode               = Enum.OrientationAlignmentMode.OneAttachment
                        flyAlignOrientation.Attachment0        = flyAttachmentGyro
                        flyAlignOrientation.MaxTorque          = math.huge
                        flyAlignOrientation.MaxAngularVelocity = math.huge
                        flyAlignOrientation.Responsiveness     = 200
                        flyAlignOrientation.RigidityEnabled    = false
                        flyAlignOrientation.Parent             = rp
                    end
                end
                return
            end

            if flyAlignOrientation then
                flyAlignOrientation.CFrame = CFrame.new(Vector3.new(), camera.CFrame.LookVector)
            end
            local moveDir = Vector3.new()
            if UserInputService:IsKeyDown(Enum.KeyCode.W)           then moveDir = moveDir + camera.CFrame.LookVector * Vector3.new(1,0,1) end
            if UserInputService:IsKeyDown(Enum.KeyCode.S)           then moveDir = moveDir - camera.CFrame.LookVector * Vector3.new(1,0,1) end
            if UserInputService:IsKeyDown(Enum.KeyCode.A)           then moveDir = moveDir - camera.CFrame.RightVector end
            if UserInputService:IsKeyDown(Enum.KeyCode.D)           then moveDir = moveDir + camera.CFrame.RightVector end
            if UserInputService:IsKeyDown(Enum.KeyCode.Space)       then moveDir = moveDir + Vector3.new(0,1,0) end
            if UserInputService:IsKeyDown(Enum.KeyCode.LeftControl) then moveDir = moveDir - Vector3.new(0,1,0) end
            if moveDir.X == 0 and moveDir.Z == 0 and humanoid then
                local md = humanoid.MoveDirection
                if md.Magnitude > 0 then
                    moveDir = Vector3.new(md.X, moveDir.Y, md.Z)
                end
            end
            if UserInputService:IsKeyDown(Enum.KeyCode.ButtonA) then
                moveDir = moveDir + Vector3.new(0, 1, 0)
            end
            if moveDir.Magnitude > 0 then moveDir = moveDir.Unit end
            moveDir = flyGroundCheck(rootPart, char, moveDir)
            if flyLinearVelocity then flyLinearVelocity.VectorVelocity = moveDir * FLY_SPEED end
        end)
        table.insert(flyConnections, moveConn)

        local respawnConn = localPlayer.CharacterAdded:Connect(function()
            if not toggles.fly then return end
            task.wait(0.25)
            if toggles.fly then toggleFly(true) end
        end)
        table.insert(flyConnections, respawnConn)

        if humanoid then
            table.insert(flyConnections, humanoid.Died:Connect(function()
                flyActive = false
                if flyLinearVelocity   then pcall(function() flyLinearVelocity:Destroy()   end); flyLinearVelocity   = nil end
                if flyAlignOrientation then pcall(function() flyAlignOrientation:Destroy() end); flyAlignOrientation = nil end
                if flyAttachment       then pcall(function() flyAttachment:Destroy()       end); flyAttachment       = nil end
                if flyAttachmentGyro   then pcall(function() flyAttachmentGyro:Destroy()   end); flyAttachmentGyro   = nil end
                _flyGroundFilterChar = nil
            end))
        end
    else
        flyActive = false
        for _, conn in ipairs(flyConnections) do conn:Disconnect() end
        flyConnections = {}
        if flyLinearVelocity   then flyLinearVelocity:Destroy();   flyLinearVelocity   = nil end
        if flyAlignOrientation then flyAlignOrientation:Destroy(); flyAlignOrientation = nil end
        if flyAttachment       then flyAttachment:Destroy();       flyAttachment       = nil end
        if flyAttachmentGyro   then flyAttachmentGyro:Destroy();   flyAttachmentGyro   = nil end
        local char = localPlayer.Character
        if char then
            local humanoid = char:FindFirstChild("Humanoid")
            if humanoid then
                humanoid.PlatformStand = flyOrigPlatformStand ~= nil
                    and flyOrigPlatformStand or false
            end
        end
        flyOrigPlatformStand = nil
        _flyGroundFilterChar = nil
        syncUndergroundRecovery()
    end
end

local vsOrigMaxSpeed = nil
local vsOrigTorque   = nil

local function disableVehicleSpeed()
    if vehiclespeedSeat and vehiclespeedSeat.Parent then
        pcall(function()
            if vsOrigMaxSpeed ~= nil then vehiclespeedSeat.MaxSpeed = vsOrigMaxSpeed end
            if vsOrigTorque   ~= nil then vehiclespeedSeat.Torque   = vsOrigTorque   end
        end)
    end
    vsOrigMaxSpeed      = nil
    vsOrigTorque        = nil
    vehiclespeedActive  = false
    vehiclespeedVehicle = nil
    vehiclespeedSeat    = nil
    
    -- FIX: Only disconnect the active session connections here (occupant). 
    -- Do NOT touch vehiclespeedMonitorConns so we keep listening for the next seat.
    for _, conn in ipairs(vehiclespeedActiveConns) do conn:Disconnect() end
    vehiclespeedActiveConns = {}
    
    if vehiclespeedLinearVelocity then vehiclespeedLinearVelocity:Destroy(); vehiclespeedLinearVelocity = nil end
    if vehiclespeedAttachment     then vehiclespeedAttachment:Destroy();     vehiclespeedAttachment     = nil end
    if vehiclespeedHeartbeat      then vehiclespeedHeartbeat:Disconnect();   vehiclespeedHeartbeat      = nil end
end

local function enableVehicleSpeed(seat)
    if not seat or not seat:IsA("VehicleSeat") then return end
    if not seat.Occupant or seat.Occupant.Parent ~= localPlayer.Character then return end

    if vehiclespeedActive and vehiclespeedSeat == seat then return end

    disableVehicleSpeed()

    vsOrigMaxSpeed = seat.MaxSpeed
    vsOrigTorque   = seat.Torque

    pcall(function()
        local speedRatio = VEHICLE_SPEED / math.max(vsOrigMaxSpeed, 10)
        seat.MaxSpeed = VEHICLE_SPEED
        seat.Torque   = vsOrigTorque * math.max(speedRatio, 1.5)
    end)

    vehiclespeedActive = true
    vehiclespeedSeat   = seat
    syncUndergroundRecovery()

    local occupantConn
    occupantConn = seat:GetPropertyChangedSignal("Occupant"):Connect(function()
        local occ = seat.Occupant
        if occ and occ.Parent == localPlayer.Character then return end
        disableVehicleSpeed()
    end)
    -- Push ONLY to the active session list so it is cleaned up perfectly on exit
    table.insert(vehiclespeedActiveConns, occupantConn)
end

function toggleVehiclespeed(enable)
    if enable then
        local function setupSeatMonitoring()
            local char = localPlayer.Character
            if not char then return end
            local humanoid = char:FindFirstChildWhichIsA("Humanoid")
            if not humanoid then return end
            if humanoid.SeatPart and humanoid.SeatPart:IsA("VehicleSeat") then
                enableVehicleSpeed(humanoid.SeatPart)
            end
            table.insert(vehiclespeedMonitorConns,
                humanoid:GetPropertyChangedSignal("SeatPart"):Connect(function()
                    if humanoid.SeatPart and humanoid.SeatPart:IsA("VehicleSeat") then
                        enableVehicleSpeed(humanoid.SeatPart)
                    else
                        disableVehicleSpeed()
                        syncUndergroundRecovery()
                    end
                end))
            table.insert(vehiclespeedMonitorConns,
                localPlayer.CharacterAdded:Connect(function()
                    if not toggles.vehiclespeed then disableVehicleSpeed(); return end
                    for _, conn in ipairs(vehiclespeedMonitorConns) do conn:Disconnect() end
                    vehiclespeedMonitorConns = {}
                    disableVehicleSpeed()
                    setupSeatMonitoring()
                end))
        end
        setupSeatMonitoring()
    else
        disableVehicleSpeed()
        -- Only full toggle-off clears the monitor connections
        for _, conn in ipairs(vehiclespeedMonitorConns) do conn:Disconnect() end
        vehiclespeedMonitorConns = {}
        syncUndergroundRecovery()
    end
end

local VEHICLE_FLY_GROUND_BUFFER = 3.5

local _vehicleFlyRayParams       = RaycastParams.new()
_vehicleFlyRayParams.FilterType  = Enum.RaycastFilterType.Exclude
local _vehicleFlyFilterParent    = nil
local _vehicleFlyFilterChar      = nil

local function vehicleFlyGroundCheck(vehiclePart, moveDir)
    if moveDir.Y >= 0 then return moveDir end
    local char         = localPlayer.Character
    local filterParent = vehiclePart.Parent or vehiclePart
    if _vehicleFlyFilterParent ~= filterParent or _vehicleFlyFilterChar ~= char then
        _vehicleFlyFilterParent = filterParent
        _vehicleFlyFilterChar   = char
        if char then
            _vehicleFlyRayParams.FilterDescendantsInstances = { filterParent, char }
        else
            _vehicleFlyRayParams.FilterDescendantsInstances = { filterParent }
        end
    end
    local checkDist = VEHICLE_FLY_GROUND_BUFFER + math.abs(moveDir.Y) * 0.2
    local result    = workspace:Raycast(vehiclePart.Position, Vector3.new(0, -checkDist, 0), _vehicleFlyRayParams)
    if result then return Vector3.new(moveDir.X, 0, moveDir.Z) end
    return moveDir
end

local function disableVehicleFly()
    if currentVehicleSeat and currentVehicleSeat.Parent then
        pcall(function()
            currentVehicleSeat.ThrottleFloat = vehicleFlyOrigThrottle ~= nil and vehicleFlyOrigThrottle or 0
            currentVehicleSeat.SteerFloat    = vehicleFlyOrigSteer    ~= nil and vehicleFlyOrigSteer    or 0
        end)
    end
    vehicleFlyOrigThrottle = nil
    vehicleFlyOrigSteer    = nil
    vehicleFlyActive   = false
    currentVehicle     = nil
    currentVehicleSeat = nil
    for _, conn in ipairs(vehicleFlyConnections) do conn:Disconnect() end
    vehicleFlyConnections = {}
    if vehicleLinearVelocity   then vehicleLinearVelocity:Destroy();   vehicleLinearVelocity   = nil end
    if vehicleAlignOrientation then vehicleAlignOrientation:Destroy(); vehicleAlignOrientation = nil end
    if vehicleAttachment       then vehicleAttachment:Destroy();       vehicleAttachment       = nil end
    if vehicleAttachmentGyro   then vehicleAttachmentGyro:Destroy();   vehicleAttachmentGyro   = nil end
    
    -- Clear strong references so RayParams doesn't hold the destroyed vehicle in memory.
    -- Setting FilterDescendantsInstances to {} is intentionally omitted: the table is
    -- rebuilt lazily on the next vehicleFlyGroundCheck call when a new vehicle is entered.
    _vehicleFlyFilterParent = nil
    _vehicleFlyFilterChar   = nil
end

local function enableVehicleFly(seat)
    if not seat or not seat:IsA("VehicleSeat") then return end
    if not seat.Occupant or seat.Occupant.Parent ~= localPlayer.Character then return end

    if vehicleFlyActive and currentVehicleSeat == seat then return end

    local vehiclePart = seat
    if seat.Parent and seat.Parent:IsA("Model") and seat.Parent.PrimaryPart then
        vehiclePart = seat.Parent.PrimaryPart
    end

    disableVehicleFly()

    local VF_MAX_FORCE         = 1e8
    local VF_MAX_TORQUE        = 1e7
    local VF_MAX_ANGVEL        = 1000
    local _vfConstraintLive    = false

    local function _vfCreateConstraints()
        if _vfConstraintLive then return end
        vehicleAttachment        = Instance.new("Attachment")
        vehicleAttachment.Parent = vehiclePart

        vehicleLinearVelocity                        = Instance.new("LinearVelocity")
        vehicleLinearVelocity.Attachment0            = vehicleAttachment
        vehicleLinearVelocity.MaxForce               = VF_MAX_FORCE
        vehicleLinearVelocity.VelocityConstraintMode = Enum.VelocityConstraintMode.Vector
        vehicleLinearVelocity.RelativeTo             = Enum.ActuatorRelativeTo.World
        vehicleLinearVelocity.VectorVelocity         = Vector3.new(0, 0, 0)
        vehicleLinearVelocity.Parent                 = vehiclePart

        vehicleAttachmentGyro        = Instance.new("Attachment")
        vehicleAttachmentGyro.Parent = vehiclePart

        vehicleAlignOrientation                     = Instance.new("AlignOrientation")
        vehicleAlignOrientation.Mode               = Enum.OrientationAlignmentMode.OneAttachment
        vehicleAlignOrientation.Attachment0        = vehicleAttachmentGyro
        vehicleAlignOrientation.MaxTorque          = VF_MAX_TORQUE
        vehicleAlignOrientation.MaxAngularVelocity = VF_MAX_ANGVEL
        vehicleAlignOrientation.Responsiveness     = 200
        vehicleAlignOrientation.RigidityEnabled    = false
        vehicleAlignOrientation.Parent             = vehiclePart
        _vfConstraintLive = true
    end

    local function _vfDestroyConstraints()
        if vehicleLinearVelocity   then vehicleLinearVelocity:Destroy();   vehicleLinearVelocity   = nil end
        if vehicleAlignOrientation then vehicleAlignOrientation:Destroy(); vehicleAlignOrientation = nil end
        if vehicleAttachment       then vehicleAttachment:Destroy();       vehicleAttachment       = nil end
        if vehicleAttachmentGyro   then vehicleAttachmentGyro:Destroy();   vehicleAttachmentGyro   = nil end
        _vfConstraintLive = false
    end

    currentVehicleSeat       = seat
    vehicleFlyOrigThrottle   = seat.ThrottleFloat
    vehicleFlyOrigSteer      = seat.SteerFloat
    currentVehicle           = vehiclePart
    vehicleFlyActive         = true
    syncUndergroundRecovery()

    _vfCreateConstraints()

    local occupantConn
    occupantConn = seat:GetPropertyChangedSignal("Occupant"):Connect(function()
        local occ = seat.Occupant
        if occ and occ.Parent == localPlayer.Character then return end
        disableVehicleFly()
    end)
    table.insert(vehicleFlyConnections, occupantConn)

    local vehicleAncConn
    vehicleAncConn = vehiclePart.AncestryChanged:Connect(function()
        if vehiclePart:IsDescendantOf(game) then return end
        disableVehicleFly()
    end)
    table.insert(vehicleFlyConnections, vehicleAncConn)

    local moveConn
    moveConn = RunService.Heartbeat:Connect(function()
        if not vehicleFlyActive or not currentVehicle or not currentVehicle.Parent then
            disableVehicleFly(); moveConn:Disconnect(); return
        end

        if not vehicleLinearVelocity   or not vehicleLinearVelocity.Parent
        or not vehicleAlignOrientation or not vehicleAlignOrientation.Parent then
            disableVehicleFly(); moveConn:Disconnect(); return
        end

        local moveDir = Vector3.new()
        if UserInputService:IsKeyDown(Enum.KeyCode.W) then moveDir = moveDir + camera.CFrame.LookVector  end
        if UserInputService:IsKeyDown(Enum.KeyCode.S) then moveDir = moveDir - camera.CFrame.LookVector  end
        if UserInputService:IsKeyDown(Enum.KeyCode.A) then moveDir = moveDir - camera.CFrame.RightVector end
        if UserInputService:IsKeyDown(Enum.KeyCode.D) then moveDir = moveDir + camera.CFrame.RightVector end
        if moveDir.Magnitude == 0 and currentVehicleSeat then
            local occ = currentVehicleSeat.Occupant
            if occ then
                local md = occ.MoveDirection
                if md.Magnitude > 0 then moveDir = Vector3.new(md.X, 0, md.Z) end
            end
        end

        if currentVehicleSeat and currentVehicleSeat.Parent then
            local wDown = UserInputService:IsKeyDown(Enum.KeyCode.W)
            local sDown = UserInputService:IsKeyDown(Enum.KeyCode.S)
            local aDown = UserInputService:IsKeyDown(Enum.KeyCode.A)
            local dDown = UserInputService:IsKeyDown(Enum.KeyCode.D)
            local throttle = (wDown and 1 or 0) - (sDown and 1 or 0)
            local steer    = (dDown and 1 or 0) - (aDown and 1 or 0)
            if throttle == 0 and steer == 0 and currentVehicleSeat.Occupant then
                local md = currentVehicleSeat.Occupant.MoveDirection
                throttle = md.Z ~= 0 and -md.Z or 0
                steer    = md.X ~= 0 and  md.X or 0
            end
            currentVehicleSeat.ThrottleFloat = throttle
            currentVehicleSeat.SteerFloat    = steer
        end

        if vehicleAlignOrientation then
            vehicleAlignOrientation.CFrame = camera.CFrame
        end

        if moveDir.Magnitude == 0 then
            vehicleLinearVelocity.VectorVelocity = Vector3.new(0, 0, 0)
            return
        end

        moveDir = moveDir.Unit
        moveDir = vehicleFlyGroundCheck(currentVehicle, moveDir)
        vehicleLinearVelocity.VectorVelocity = moveDir * VEHICLE_FLY_SPEED
    end)
    table.insert(vehicleFlyConnections, moveConn)
end

function toggleVehicleFly(enable)
    if enable then
        local function setupSeatMonitoring()
            local char = localPlayer.Character
            if not char then return end
            local humanoid = char:FindFirstChildWhichIsA("Humanoid")
            if not humanoid then return end
            if humanoid.SeatPart and humanoid.SeatPart:IsA("VehicleSeat") then
                enableVehicleFly(humanoid.SeatPart)
            end
            table.insert(vehicleMonitorConns,
                humanoid:GetPropertyChangedSignal("SeatPart"):Connect(function()
                    if humanoid.SeatPart and humanoid.SeatPart:IsA("VehicleSeat") then
                        enableVehicleFly(humanoid.SeatPart)
                    else
                        disableVehicleFly()
                        syncUndergroundRecovery()
                    end
                end))
            table.insert(vehicleMonitorConns,
                localPlayer.CharacterAdded:Connect(function()
                    if not toggles.vehicleFly then disableVehicleFly(); return end
                    for _, conn in ipairs(vehicleMonitorConns) do conn:Disconnect() end
                    vehicleMonitorConns = {}
                    disableVehicleFly()
                    setupSeatMonitoring()
                end))
        end
        setupSeatMonitoring()
    else
        disableVehicleFly()
        for _, conn in ipairs(vehicleMonitorConns) do conn:Disconnect() end
        vehicleMonitorConns = {}
        syncUndergroundRecovery()
    end
end

local NO_RECOIL_SENS_INITIAL    = 0.013
local NO_RECOIL_CALIB_MIN_DELTA = 1.5
local NO_RECOIL_CALIB_WARMUP    = 30
local NO_RECOIL_CALIB_ALPHA     = 0.04
local NO_RECOIL_THRESHOLD_RAD   = 0.002

local NO_RECOIL_BOOL_KEYWORDS = {
    "canfire","canshoot","canreload","isreloading","firing","reloading",
    "debounce","cooldown","fireready","shootready","isready",
    "waitingforreload","waitingforfire","onfirecd","onreloadcd",
}
local function isBoolGate(name)
    local lower = name:lower()
    for _, kw in ipairs(NO_RECOIL_BOOL_KEYWORDS) do
        if lower:find(kw, 1, true) then return true end
    end
    return false
end

local NO_RECOIL_RELOAD_ANIM_KW = {
    "reload","reloading","refill","chamber","cock","bolt","mag",
}
local function isReloadAnimName(name)
    local lower = name:lower()
    for _, kw in ipairs(NO_RECOIL_RELOAD_ANIM_KW) do
        if lower:find(kw, 1, true) then return true end
    end
    return false
end

local NO_RECOIL_COOLDOWN_KEYWORDS = {
    "cooldown","firerate","firedelay","reloadtime","reloadduration",
    "debounce","firedebt","shotdelay","rate","delay","timer","wait",
    "shootcooldown","attackcooldown","equipcooldown",
    "spread","bloom","recoil","accuracy","scatter","dispersion",
    "deviation","inaccuracy","bulletspread","aimspread","currentspread",
    "currentbloom","spreadvalue","bloomvalue","spreadfactor","spreadangle",
    "hipspread","adsbloom","walkspread","runspread","crouchspread",
}

local function nrSaveAndSet(obj, prop, newVal)
    local objSlot = nrSavedLookup[obj]
    if objSlot and objSlot[prop] then
        pcall(function() obj[prop] = newVal end)
        return
    end
    local orig; local ok = pcall(function() orig = obj[prop] end)
    if ok then
        table.insert(noRecoilSavedValues, { obj = obj, prop = prop, orig = orig })
        if not nrSavedLookup[obj] then nrSavedLookup[obj] = {} end
        nrSavedLookup[obj][prop] = true
        pcall(function() obj[prop] = newVal end)
    end
end

local NO_RECOIL_SPREAD_KEYWORDS = {
    "spread","bloom","recoil","accuracy","scatter","dispersion",
    "deviation","inaccuracy","bulletspread","aimspread","currentspread",
    "currentbloom","spreadvalue","bloomvalue","spreadfactor","spreadangle",
    "hipspread","adsbloom","walkspread","runspread","crouchspread",
}
local function isSpreadKeyword(name)
    local lower = name:lower()
    for _, kw in ipairs(NO_RECOIL_SPREAD_KEYWORDS) do
        if lower:find(kw, 1, true) then return true end
    end
    return false
end
local function isCooldownKeyword(name)
    local lower = name:lower()
    for _, kw in ipairs(NO_RECOIL_COOLDOWN_KEYWORDS) do
        if lower:find(kw, 1, true) then return true end
    end
    return false
end
local function nrAddSpreadTarget(entry)
    local key = entry.attrName or entry.prop
    local objSlot = nrSpreadLookup[entry.obj]
    if objSlot and objSlot[key] then return end
    if not nrSpreadLookup[entry.obj] then nrSpreadLookup[entry.obj] = {} end
    nrSpreadLookup[entry.obj][key] = true
    table.insert(noRecoilSpreadTargets, entry)
end

local NR_PHYSICS_MOVER_TYPES = {
    "BodyAngularVelocity", "BodyGyro", "AngularVelocity", "BodyVelocity",
}
local function isRecoilPhysicsMover(obj)
    for _, t in ipairs(NR_PHYSICS_MOVER_TYPES) do
        if obj:IsA(t) then return true end
    end
    return false
end

local function nrHookRootPart(rootPart)
    if not rootPart then return nil end
    for _, child in ipairs(rootPart:GetChildren()) do
        if isRecoilPhysicsMover(child) then pcall(function() child:Destroy() end) end
    end
    return rootPart.ChildAdded:Connect(function(child)
        if not toggles.noRecoil then return end
        if isRecoilPhysicsMover(child) then
            pcall(function() child:Destroy() end)
        end
    end)
end

local function nrScanObject(obj)
    if obj:IsA("NumberValue") or obj:IsA("IntValue") then
        local name = obj.Name
        if isSpreadKeyword(name) then
            nrSaveAndSet(obj, "Value", 0)
            nrAddSpreadTarget({ obj = obj, prop = "Value" })
        elseif isCooldownKeyword(name) then
            nrSaveAndSet(obj, "Value", 0)
        end
    elseif obj:IsA("BoolValue") then
        if isBoolGate(obj.Name) then
            local lower = obj.Name:lower()
            local forceTrue = (lower:find("^can") ~= nil) or (lower:find("ready") ~= nil) or (lower:find("equipped") ~= nil)
            nrSaveAndSet(obj, "Value", forceTrue)
            nrAddSpreadTarget({ obj = obj, prop = "Value", boolForce = forceTrue })
        end
    end
end

local function nrScanAttributesDeep(container)
    local function scanAttrs(obj)
        pcall(function()
            for attrName, attrVal in pairs(obj:GetAttributes()) do
                if type(attrVal) == "number" then
                    local isSpread   = isSpreadKeyword(attrName)
                    local isCooldown = not isSpread and isCooldownKeyword(attrName)
                    if not isSpread and not isCooldown then continue end
                    local key = "_attr_" .. obj:GetFullName() .. "_" .. attrName
                    local objSlot = nrSavedLookup[obj]
                    if not (objSlot and objSlot[attrName]) then
                        table.insert(noRecoilSavedValues, {
                            obj = obj, prop = key, orig = attrVal,
                            isAttr = true, attrName = attrName,
                        })
                        if not nrSavedLookup[obj] then nrSavedLookup[obj] = {} end
                        nrSavedLookup[obj][attrName] = true
                    end
                    pcall(function() obj:SetAttribute(attrName, 0) end)
                    if isSpread then
                        nrAddSpreadTarget({ obj = obj, prop = key, isAttr = true, attrName = attrName })
                    end
                elseif type(attrVal) == "boolean" then
                    if isBoolGate(attrName) then
                        local lower = attrName:lower()
                        local forced = (lower:find("^can") ~= nil) or (lower:find("ready") ~= nil) or (lower:find("equipped") ~= nil)
                        local key = "_attrb_" .. obj:GetFullName() .. "_" .. attrName
                        local objSlot = nrSavedLookup[obj]
                        if not (objSlot and objSlot[attrName]) then
                            table.insert(noRecoilSavedValues, {
                                obj = obj, prop = key, orig = attrVal,
                                isAttr = true, attrName = attrName,
                            })
                            if not nrSavedLookup[obj] then nrSavedLookup[obj] = {} end
                            nrSavedLookup[obj][attrName] = true
                        end
                        pcall(function() obj:SetAttribute(attrName, forced) end)
                        nrAddSpreadTarget({ obj = obj, prop = key, isAttr = true,
                            attrName = attrName, boolForce = forced })
                    end
                end
            end
        end)
    end
    scanAttrs(container)
    for _, desc in ipairs(container:GetDescendants()) do scanAttrs(desc) end
end

local function nrSkipReloadAnims(char)
    if not char then return end
    local animator = char:FindFirstChildWhichIsA("Animator", true)
    if not animator then return end
    pcall(function()
        for _, track in ipairs(animator:GetPlayingAnimationTracks()) do
            if isReloadAnimName(track.Name) then
                track.TimePosition = track.Length * 0.95
            end
        end
    end)
end

local function toolHasSpread(tool)
    for _, desc in ipairs(tool:GetDescendants()) do
        if (desc:IsA("NumberValue") or desc:IsA("IntValue")) and isSpreadKeyword(desc.Name) then
            return true
        end
    end
    local function hasSpreadAttr(obj)
        local ok, attrs = pcall(function() return obj:GetAttributes() end)
        if not ok then return false end
        for attrName, attrVal in pairs(attrs) do
            if type(attrVal) == "number" and isSpreadKeyword(attrName) then return true end
        end
        return false
    end
    if hasSpreadAttr(tool) then return true end
    for _, desc in ipairs(tool:GetDescendants()) do
        if hasSpreadAttr(desc) then return true end
    end
    return false
end

local function nrApplyToTool(tool)
    if not tool or not tool:IsA("Tool") then return end
    if nrAppliedTools[tool] then return end
    nrAppliedTools[tool] = true

    if _nrToolConnSlots[tool] then
        for _, c in ipairs(_nrToolConnSlots[tool]) do c:Disconnect() end
        _nrToolConnSlots[tool] = nil
    end

    local isRanged = toolHasSpread(tool)
    local slot = {}  

    local function releaseSlot()
        nrAppliedTools[tool] = nil
        if _nrToolConnSlots[tool] then
            for _, c in ipairs(_nrToolConnSlots[tool]) do c:Disconnect() end
            _nrToolConnSlots[tool] = nil
        end
    end

    if not isRanged then
        for _, desc in ipairs(tool:GetDescendants()) do
            if desc:IsA("NumberValue") or desc:IsA("IntValue") then
                if isCooldownKeyword(desc.Name) then
                    pcall(function() desc.Value = 0 end)
                end
            elseif desc:IsA("BoolValue") then
                if isBoolGate(desc.Name) then
                    local lower = desc.Name:lower()
                    local forceTrue = (lower:find("^can") ~= nil) or (lower:find("ready") ~= nil)
                    pcall(function() desc.Value = forceTrue end)
                end
            end
        end
        pcall(function()
            for attrName, attrVal in pairs(tool:GetAttributes()) do
                if type(attrVal) == "number" and isCooldownKeyword(attrName) then
                    pcall(function() tool:SetAttribute(attrName, 0) end)
                elseif type(attrVal) == "boolean" and isBoolGate(attrName) then
                    local lower = attrName:lower()
                    local forceTrue = (lower:find("^can") ~= nil) or (lower:find("ready") ~= nil)
                    pcall(function() tool:SetAttribute(attrName, forceTrue) end)
                end
            end
        end)
        local meleeAncConn
        meleeAncConn = tool.AncestryChanged:Connect(function()
            if tool:IsDescendantOf(game) then return end
            releaseSlot()
        end)
        slot[#slot+1] = meleeAncConn
        _nrToolConnSlots[tool] = slot
        return
    end

    for _, desc in ipairs(tool:GetDescendants()) do nrScanObject(desc) end
    nrScanAttributesDeep(tool)

    local descConn = tool.DescendantAdded:Connect(function(desc)
        if not toggles.noRecoil then return end
        nrScanObject(desc)
        if desc:IsA("Configuration") or desc:IsA("Folder") then
            nrScanAttributesDeep(desc)
        end
    end)
    slot[#slot+1] = descConn

    local toolRemovingConn = tool.DescendantRemoving:Connect(function(obj)
        if nrSavedLookup[obj] then nrSavedLookup[obj] = nil end
        local n = #noRecoilSavedValues
        local i = 1
        while i <= n do
            if noRecoilSavedValues[i].obj == obj then
                noRecoilSavedValues[i] = noRecoilSavedValues[n]
                noRecoilSavedValues[n] = nil
                n = n - 1
            else
                i = i + 1
            end
        end
    end)
    slot[#slot+1] = toolRemovingConn

    local activateConn = tool.Activated:Connect(function()
        if not toggles.noRecoil then return end
        local char = localPlayer.Character
        if char then nrSkipReloadAnims(char) end
    end)
    slot[#slot+1] = activateConn

    local ancestryConn
    ancestryConn = tool.AncestryChanged:Connect(function()
        if tool:IsDescendantOf(game) then return end
        releaseSlot()
        local liveTargets = {}
        for _, t in ipairs(noRecoilSpreadTargets) do
            if t.obj and t.obj.Parent then liveTargets[#liveTargets+1] = t end
        end
        noRecoilSpreadTargets = liveTargets
        local liveSaved = {}
        for _, s in ipairs(noRecoilSavedValues) do
            if s.obj and s.obj.Parent then liveSaved[#liveSaved+1] = s end
        end
        noRecoilSavedValues = liveSaved
        nrSavedLookup  = {}
        nrSpreadLookup = {}
        for _, s in ipairs(noRecoilSavedValues)   do
            if not nrSavedLookup[s.obj]  then nrSavedLookup[s.obj]  = {} end
            nrSavedLookup[s.obj][s.attrName or s.prop] = true
        end
        for _, t in ipairs(noRecoilSpreadTargets) do
            if not nrSpreadLookup[t.obj] then nrSpreadLookup[t.obj] = {} end
            nrSpreadLookup[t.obj][t.attrName or t.prop] = true
        end
    end)
    slot[#slot+1] = ancestryConn

    _nrToolConnSlots[tool] = slot
end

local function nrScanCharacter(char)
    if not char then return end
    for _, desc in ipairs(char:GetDescendants()) do
        if desc:IsA("NumberValue") or desc:IsA("IntValue") then
            if isSpreadKeyword(desc.Name) then
                nrSaveAndSet(desc, "Value", 0)
                nrAddSpreadTarget({ obj = desc, prop = "Value" })
            end
        end
    end
    local charDescConn = char.DescendantAdded:Connect(function(desc)
        if not toggles.noRecoil then return end
        if (desc:IsA("NumberValue") or desc:IsA("IntValue"))
            and isSpreadKeyword(desc.Name) then
            nrSaveAndSet(desc, "Value", 0)
            nrAddSpreadTarget({ obj = desc, prop = "Value" })
        end
    end)
    table.insert(noRecoilCharConns, charDescConn)

    local charRemovConn = char.DescendantRemoving:Connect(function(obj)
        if nrSavedLookup[obj] then nrSavedLookup[obj] = nil end
        local n = #noRecoilSavedValues
        local i = 1
        while i <= n do
            if noRecoilSavedValues[i].obj == obj then
                noRecoilSavedValues[i] = noRecoilSavedValues[n]
                noRecoilSavedValues[n] = nil
                n = n - 1
            else
                i = i + 1
            end
        end
    end)
    table.insert(noRecoilCharConns, charRemovConn)
    local rootPart = char:FindFirstChild("HumanoidRootPart")
    if rootPart then
        local rootConn = nrHookRootPart(rootPart)
        if rootConn then table.insert(noRecoilCharConns, rootConn) end
    else
        local waitConn
        waitConn = char.ChildAdded:Connect(function(child)
            if child.Name == "HumanoidRootPart" then
                waitConn:Disconnect()
                local rootConn = nrHookRootPart(child)
                if rootConn then table.insert(noRecoilCharConns, rootConn) end
            end
        end)
        table.insert(noRecoilCharConns, waitConn)
    end
end

function toggleNoRecoil(enable)
    if enable then
        noRecoilPrevCF        = camera.CFrame
        noRecoilDriftAccum    = 0
        noRecoilSensLive      = NO_RECOIL_SENS_INITIAL
        noRecoilCalibSamples  = 0

        local char = localPlayer.Character
        if char then
            nrScanCharacter(char)
            for _, child in ipairs(char:GetChildren()) do
                if child:IsA("Tool") then nrApplyToTool(child) end
            end
            if noRecoilCharToolConn then noRecoilCharToolConn:Disconnect() end
            noRecoilCharToolConn = char.ChildAdded:Connect(function(child)
                if not toggles.noRecoil then return end
                if child:IsA("Tool") then nrApplyToTool(child) end
            end)
        end

        local charRemovingConn = localPlayer.CharacterRemoving:Connect(function(oldChar)
            if not toggles.noRecoil then return end
            for _, c in ipairs(noRecoilCharConns) do c:Disconnect() end
            noRecoilCharConns = {}
            for _, s in ipairs(noRecoilSavedValues) do
                if s.obj and s.obj.Parent then
                    if s.isAttr then
                        pcall(function() s.obj:SetAttribute(s.attrName, s.orig) end)
                    else
                        pcall(function() s.obj[s.prop] = s.orig end)
                    end
                end
            end
            noRecoilSavedValues   = {}
            noRecoilSpreadTargets = {}
            nrSavedLookup  = {}  
            nrSpreadLookup = {}
            nrAppliedTools = {}
            local currentChar = localPlayer.Character
            if currentChar and currentChar ~= oldChar then
                task.defer(function()
                    if not toggles.noRecoil then return end
                    nrScanCharacter(currentChar)
                    for _, child in ipairs(currentChar:GetChildren()) do
                        if child:IsA("Tool") then nrApplyToTool(child) end
                    end
                end)
            end
        end)
        table.insert(noRecoilToolConns, charRemovingConn)

        local respawnConn = localPlayer.CharacterAdded:Connect(function(newChar)
            if not toggles.noRecoil then return end
            for _, c in ipairs(noRecoilCharConns) do c:Disconnect() end
            noRecoilCharConns = {}
            task.wait(0.1)
            if not toggles.noRecoil then return end
            noRecoilPrevCF       = camera.CFrame
            noRecoilDriftAccum   = 0
            noRecoilSensLive     = NO_RECOIL_SENS_INITIAL
            noRecoilCalibSamples = 0
            nrScanCharacter(newChar)
            for _, child in ipairs(newChar:GetChildren()) do
                if child:IsA("Tool") then nrApplyToTool(child) end
            end
            if noRecoilCharToolConn then noRecoilCharToolConn:Disconnect() end
            noRecoilCharToolConn = newChar.ChildAdded:Connect(function(child)
                if not toggles.noRecoil then return end
                if child:IsA("Tool") then nrApplyToTool(child) end
            end)
        end)
        table.insert(noRecoilToolConns, respawnConn)

        local nrSpreadFrame = 0
        noRecoilHeartbeatConn = RunService.Heartbeat:Connect(function()
            if not toggles.noRecoil then return end
            nrSpreadFrame = nrSpreadFrame + 1

            local current    = camera.CFrame
            local mouseDelta = UserInputService:GetMouseDelta()
            local isFiring   = UserInputService:IsMouseButtonPressed(Enum.UserInputType.MouseButton1)

            if noRecoilPrevCF then
                local prevPitch   = math.asin(math.clamp(noRecoilPrevCF.LookVector.Y, -1, 1))
                local actualPitch = math.asin(math.clamp(current.LookVector.Y,        -1, 1))
                local pitchDelta  = actualPitch - prevPitch

                if not isFiring then
                    local absMouseY = math.abs(mouseDelta.Y)
                    if absMouseY >= NO_RECOIL_CALIB_MIN_DELTA then
                        local measuredSens = -pitchDelta / mouseDelta.Y
                        if measuredSens > 0.001 and measuredSens < 0.08 then
                            if noRecoilCalibSamples < NO_RECOIL_CALIB_WARMUP then
                                noRecoilSensLive    = (noRecoilSensLive * noRecoilCalibSamples
                                                      + measuredSens) / (noRecoilCalibSamples + 1)
                                noRecoilCalibSamples = noRecoilCalibSamples + 1
                            else
                                noRecoilSensLive = noRecoilSensLive * (1 - NO_RECOIL_CALIB_ALPHA)
                                                 + measuredSens      * NO_RECOIL_CALIB_ALPHA
                            end
                        end
                    end
                end

                if isFiring then
                    local expectedPitch = -mouseDelta.Y * noRecoilSensLive
                    local pitchExcess   = pitchDelta - expectedPitch
                    pitchExcess = math.clamp(pitchExcess, 0, 0.15)

                    if pitchExcess > NO_RECOIL_THRESHOLD_RAD then
                        noRecoilDriftAccum = 0
                        local corrected = CFrame.new(current.Position)
                            * CFrame.fromAxisAngle(current.RightVector, -pitchExcess)
                            * current.Rotation
                        camera.CFrame = corrected
                        current = corrected
                    elseif pitchExcess > 0 then
                        noRecoilDriftAccum = noRecoilDriftAccum + pitchExcess
                        if noRecoilDriftAccum > NO_RECOIL_THRESHOLD_RAD then
                            local correction   = noRecoilDriftAccum
                            noRecoilDriftAccum = 0
                            local corrected = CFrame.new(current.Position)
                                * CFrame.fromAxisAngle(current.RightVector, -correction)
                                * current.Rotation
                            camera.CFrame = corrected
                            current = corrected
                        end
                    else
                        noRecoilDriftAccum = math.max(0, noRecoilDriftAccum - 0.0008)
                    end
                else
                    noRecoilDriftAccum = math.max(0, noRecoilDriftAccum - 0.002)
                end
            end
            noRecoilPrevCF = current

            if #noRecoilSpreadTargets > 0 then
                local shouldWrite = isFiring or (nrSpreadFrame % 6 == 0)
                local i = 1
                while i <= #noRecoilSpreadTargets do
                    local t = noRecoilSpreadTargets[i]
                    if not (t.obj and t.obj.Parent) then
                        local objSlot = nrSpreadLookup[t.obj]
                        if objSlot then objSlot[t.attrName or t.prop] = nil end  
                        noRecoilSpreadTargets[i] = noRecoilSpreadTargets[#noRecoilSpreadTargets]
                        noRecoilSpreadTargets[#noRecoilSpreadTargets] = nil
                    else
                        if shouldWrite then
                            if t.boolForce ~= nil then
                                if t.isAttr then
                                    t.obj:SetAttribute(t.attrName, t.boolForce)
                                else
                                    if t.obj.Value ~= t.boolForce then
                                        t.obj.Value = t.boolForce
                                    end
                                end
                            elseif t.isAttr then
                                t.obj:SetAttribute(t.attrName, 0)
                            else
                                if t.obj[t.prop] ~= 0 then t.obj[t.prop] = 0 end
                            end
                        end
                        i = i + 1
                    end
                end
            end
        end)

    else
        if noRecoilHeartbeatConn then
            noRecoilHeartbeatConn:Disconnect()
            noRecoilHeartbeatConn = nil
        end
        noRecoilPrevCF       = nil
        noRecoilDriftAccum   = 0
        noRecoilSensLive     = NO_RECOIL_SENS_INITIAL
        noRecoilCalibSamples = 0

        for _, c in ipairs(noRecoilToolConns) do c:Disconnect() end
        noRecoilToolConns = {}
        if noRecoilCharToolConn then noRecoilCharToolConn:Disconnect(); noRecoilCharToolConn = nil end
        for _, c in ipairs(noRecoilCharConns) do c:Disconnect() end
        noRecoilCharConns = {}
        for _, slot in pairs(_nrToolConnSlots) do
            for _, c in ipairs(slot) do c:Disconnect() end
        end
        _nrToolConnSlots = {}
        nrAppliedTools = {}
        noRecoilSpreadTargets = {}
        nrSpreadLookup = {}

        for _, s in ipairs(noRecoilSavedValues) do
            if s.obj and s.obj.Parent then
                if s.isAttr then
                    pcall(function() s.obj:SetAttribute(s.attrName, s.orig) end)
                else
                    pcall(function() s.obj[s.prop] = s.orig end)
                end
            end
        end
        noRecoilSavedValues = {}
        nrSavedLookup = {}
    end
end

local function applyInstantInteract(prompt)
    if not prompt:IsA("ProximityPrompt") then return end
    if not origHoldDurations[prompt] then
        origHoldDurations[prompt] = prompt.HoldDuration
    end
    prompt.HoldDuration = 0
end

function toggleInstantInteract(enable)
    if enable then
        for _, desc in ipairs(Workspace:GetDescendants()) do
            applyInstantInteract(desc)
        end
        local iiPendingQueue = {}
        local iiDrainQueued  = false
        local function iiDrainQueue()
            iiDrainQueued = false
            if not toggles.instantInteract then iiPendingQueue = {}; return end
            for i = 1, #iiPendingQueue do
                applyInstantInteract(iiPendingQueue[i])
            end
            iiPendingQueue = {}
        end
        table.insert(instantInteractConns, Workspace.DescendantAdded:Connect(function(desc)
            if not toggles.instantInteract then return end
            if not desc:IsA("ProximityPrompt") then return end
            iiPendingQueue[#iiPendingQueue + 1] = desc
            if not iiDrainQueued then
                iiDrainQueued = true
                task.defer(iiDrainQueue)
            end
        end))

        table.insert(instantInteractConns, Workspace.DescendantRemoving:Connect(function(obj)
            if origHoldDurations[obj] then origHoldDurations[obj] = nil; return end
            if next(origHoldDurations) and (obj:IsA("Model") or obj:IsA("Folder")) then
                for prompt in pairs(origHoldDurations) do
                    if prompt:IsDescendantOf(obj) then origHoldDurations[prompt] = nil end
                end
            end
        end))
        table.insert(instantInteractConns, Workspace.ChildRemoved:Connect(function(removed)
            if not (removed:IsA("Model") or removed:IsA("Folder")) then return end
            for prompt in pairs(origHoldDurations) do
                if prompt:IsDescendantOf(removed) then origHoldDurations[prompt] = nil end
            end
        end))

        table.insert(instantInteractConns, localPlayer.CharacterAdded:Connect(function()
            if not toggles.instantInteract then return end
            for prompt, orig in pairs(origHoldDurations) do
                if prompt and prompt.Parent then
                    prompt.HoldDuration = orig
                end
            end
            origHoldDurations = {}  
            for _, desc in ipairs(Workspace:GetDescendants()) do
                applyInstantInteract(desc)
            end
        end))
    else
        for _, conn in ipairs(instantInteractConns) do conn:Disconnect() end
        instantInteractConns = {}
        for prompt, orig in pairs(origHoldDurations) do
            if prompt and prompt.Parent then
                prompt.HoldDuration = orig
            end
        end
        origHoldDurations = {}  
    end
end

    local restoreCorner = Instance.new("UICorner")
    restoreCorner.CornerRadius = UDim.new(0,16)
    restoreCorner.Parent       = restoreButton

    table.insert(uiConnections, restoreButton.InputBegan:Connect(function(input)
        if input.UserInputType ~= Enum.UserInputType.MouseButton1
            and input.UserInputType ~= Enum.UserInputType.Touch then return end

        local startInputPos = Vector2.new(input.Position.X, input.Position.Y)
        local startAbsPos   = restoreButton.AbsolutePosition
        local moved         = false
        local movedConn, endedConn

        movedConn = UserInputService.InputChanged:Connect(function(changed)
            if changed.UserInputType ~= Enum.UserInputType.MouseMovement
                and changed.UserInputType ~= Enum.UserInputType.Touch then return end
            local delta = Vector2.new(changed.Position.X, changed.Position.Y) - startInputPos
            if delta.Magnitude > 5 then moved = true end
            local newX = math.clamp(startAbsPos.X + delta.X, 0, screenGui.AbsoluteSize.X - restoreButton.AbsoluteSize.X)
            local newY = math.clamp(startAbsPos.Y + delta.Y, 0, screenGui.AbsoluteSize.Y - restoreButton.AbsoluteSize.Y)
            restoreButton.Position = UDim2.new(0, newX, 0, newY)
        end)

        endedConn = UserInputService.InputEnded:Connect(function(ended)
            if ended ~= input then return end
            movedConn:Disconnect()
            endedConn:Disconnect()
            restoreButtonPos = restoreButton.Position
            if not moved and isMinimized then
                mainFrame.Position    = UDim2.new(0.5,-150,0.5,-185)
                mainFrame.Visible     = true
                restoreButton.Visible = false
                isMinimized           = false
                guiVisible            = true
            end
        end)
    end))
end

local function setupUIEvents()
    table.insert(uiConnections, minimizeButton.MouseButton1Click:Connect(function()
        restoreButton.Position = restoreButtonPos
        restoreButton.Visible  = true
        mainFrame.Visible      = false
        isMinimized            = true
    end))

    local closePending = false
    table.insert(uiConnections, closeButton.MouseButton1Click:Connect(function()
        if closePending then
            cleanupAll()
        else
            closePending = true
            closeButton.BackgroundColor3 = Color3.fromRGB(230,100,100)
            task.wait(2)
            closePending = false
            closeButton.BackgroundColor3 = Color3.fromRGB(190,50,50)
        end
    end))

    table.insert(uiConnections, UserInputService.InputBegan:Connect(function(input, gameProcessed)
        local isHotkey = (input.KeyCode == Enum.KeyCode.Insert)
            or (input.KeyCode == Enum.KeyCode.RightControl and not gameProcessed)
        if not isHotkey then return end
        if isMinimized then
            restoreButton.Visible = false
            mainFrame.Visible     = true
            isMinimized           = false
            guiVisible            = true
        else
            guiVisible        = not guiVisible
            mainFrame.Visible = guiVisible
        end
    end))
end

function cleanupAll()
    stopGlobalDistanceUpdater()
    stopUndergroundRecovery()

    local toDestroy = {}
    for inst in pairs(espInstances) do toDestroy[#toDestroy+1] = inst end
    for _, inst in ipairs(toDestroy) do destroyESPInstance(inst) end
    espInstances = {}

    for _, typeConns in pairs(espTypeConnections) do
        for _, conn in ipairs(typeConns) do conn:Disconnect() end
    end
    espTypeConnections = { players={}, items={}, npcs={}, helditems={} }
    for _, conn in ipairs(localCharESPConns) do conn:Disconnect() end
    localCharESPConns = {}

    local toUntrackPlayers = {}
    for player in pairs(playerTrackers) do toUntrackPlayers[#toUntrackPlayers+1] = player end
    for _, player in ipairs(toUntrackPlayers) do untrackPlayer(player) end

    local toUntrackHeld = {}
    for player in pairs(heldItemTrackers) do toUntrackHeld[#toUntrackHeld+1] = player end
    for _, player in ipairs(toUntrackHeld) do untrackPlayerHeldItems(player) end

    playerTrackers   = {}
    heldItemTrackers = {}

    if noclipSteppedConn then noclipSteppedConn:Disconnect(); noclipSteppedConn = nil end
    if noclipCharDescConn then noclipCharDescConn:Disconnect(); noclipCharDescConn = nil end
    for _, conn in ipairs(noclipPartConns)  do conn:Disconnect() end; noclipPartConns  = {}
    for _, conn in ipairs(noclipConnections) do conn:Disconnect() end; noclipConnections = {}
    noclipPartsCache = {}
    for part, original in pairs(noclipOriginalStates) do
        if part and part.Parent then part.CanCollide = original end
    end
    noclipOriginalStates = {}

    flyActive = false
    for _, conn in ipairs(flyConnections) do conn:Disconnect() end
    flyConnections = {}
    if flyLinearVelocity   then flyLinearVelocity:Destroy();   flyLinearVelocity   = nil end
    if flyAlignOrientation then flyAlignOrientation:Destroy(); flyAlignOrientation = nil end
    if flyAttachment       then flyAttachment:Destroy();       flyAttachment       = nil end
    if flyAttachmentGyro   then flyAttachmentGyro:Destroy();   flyAttachmentGyro   = nil end
    do
        local char = localPlayer.Character
        if char then
            local h = char:FindFirstChild("Humanoid")
            if h then
                h.PlatformStand = flyOrigPlatformStand ~= nil and flyOrigPlatformStand or false
            end
        end
        flyOrigPlatformStand = nil
    end
    _flyGroundFilterChar = nil  -- release character reference; filter rebuilt on next fly enable

    for _, conn in ipairs(tpToolConns) do conn:Disconnect() end
    tpToolConns = {}
    for _, conn in ipairs(_tpObjConns) do conn:Disconnect() end
    _tpObjConns = {}
    if toggles.tpToolObject then
        pcall(function() toggles.tpToolObject:Destroy() end)
        toggles.tpToolObject = nil
    end

    disableVehicleFly()
    for _, conn in ipairs(vehicleMonitorConns) do conn:Disconnect() end
    vehicleMonitorConns = {}

    disableVehicleSpeed()  -- clears vehiclespeedActiveConns + restores seat properties
    for _, conn in ipairs(vehiclespeedActiveConns) do conn:Disconnect() end  -- FIX: belt-and-suspenders clear in case disableVehicleSpeed was already called while not seated
    vehiclespeedActiveConns = {}
    for _, conn in ipairs(vehiclespeedMonitorConns) do conn:Disconnect() end
    vehiclespeedMonitorConns = {}

    if _wsCharAddedConn then _wsCharAddedConn:Disconnect(); _wsCharAddedConn = nil end
    if _wsHumPropConn   then _wsHumPropConn:Disconnect();   _wsHumPropConn   = nil end
    if _wsHumAncConn    then _wsHumAncConn:Disconnect();    _wsHumAncConn    = nil end
    if _wsCharDescConn  then _wsCharDescConn:Disconnect();  _wsCharDescConn  = nil end
    if _wsCharAncConn   then _wsCharAncConn:Disconnect();   _wsCharAncConn   = nil end
    for humanoid, original in pairs(walkspeedOriginalValues) do
        if humanoid and humanoid.Parent then humanoid.WalkSpeed = original end
    end
    walkspeedOriginalValues = {}

    if fpsBoostProperties.QualityLevel ~= nil then
        pcall(function()
            settings().Rendering.QualityLevel = fpsBoostProperties.QualityLevel
        end)
        fpsBoostProperties.QualityLevel = nil
    end

    if originalLightingProperties then
        for prop, val in pairs(originalLightingProperties) do
            pcall(function() Lighting[prop] = val end)
        end
        originalLightingProperties = nil
    end

    if fpsBoostProperties.Terrain then
        pcall(function() Workspace.Terrain.WaterWaveSize     = fpsBoostProperties.Terrain.WaterWaveSize     end)
        pcall(function() Workspace.Terrain.WaterWaveSpeed    = fpsBoostProperties.Terrain.WaterWaveSpeed    end)
        pcall(function() Workspace.Terrain.WaterReflectance  = fpsBoostProperties.Terrain.WaterReflectance  end)
        pcall(function() Workspace.Terrain.WaterTransparency = fpsBoostProperties.Terrain.WaterTransparency end)
    end
    fpsBoostProperties = {}

    if fpsBoostData.descendantConn  then fpsBoostData.descendantConn:Disconnect();  fpsBoostData.descendantConn  = nil end
    if fpsBoostData.charAddedConn   then fpsBoostData.charAddedConn:Disconnect();   fpsBoostData.charAddedConn   = nil end
    if fpsBoostData.removingConn    then fpsBoostData.removingConn:Disconnect();    fpsBoostData.removingConn    = nil end
    if fpsBoostData.mapRemovedConn  then fpsBoostData.mapRemovedConn:Disconnect();  fpsBoostData.mapRemovedConn  = nil end
    restoreFPSBoostData()

    if noRecoilHeartbeatConn then
        noRecoilHeartbeatConn:Disconnect()
        noRecoilHeartbeatConn = nil
    end
    noRecoilPrevCF       = nil
    noRecoilDriftAccum   = 0
    noRecoilSensLive     = NO_RECOIL_SENS_INITIAL
    noRecoilCalibSamples = 0
    for _, c in ipairs(noRecoilToolConns) do c:Disconnect() end
    noRecoilToolConns = {}
    if noRecoilCharToolConn then noRecoilCharToolConn:Disconnect(); noRecoilCharToolConn = nil end
    for _, c in ipairs(noRecoilCharConns) do c:Disconnect() end
    noRecoilCharConns   = {}
    for _, slot in pairs(_nrToolConnSlots) do
        for _, c in ipairs(slot) do c:Disconnect() end
    end
    _nrToolConnSlots = {}
    nrAppliedTools = {}
    noRecoilSpreadTargets = {}
    nrSpreadLookup = {}
    for _, s in ipairs(noRecoilSavedValues) do
        if s.obj and s.obj.Parent then
            if s.isAttr then
                pcall(function() s.obj:SetAttribute(s.attrName, s.orig) end)
            else
                pcall(function() s.obj[s.prop] = s.orig end)
            end
        end
    end
    noRecoilSavedValues = {}
    nrSavedLookup = {}
    
    for _, conn in ipairs(instantInteractConns) do conn:Disconnect() end
    instantInteractConns = {}
    for prompt, orig in pairs(origHoldDurations) do
        if prompt and prompt.Parent then
            prompt.HoldDuration = orig
        end
    end
    origHoldDurations = {} 

    for _, conn in ipairs(uiConnections) do
        if conn and conn.Connected then conn:Disconnect() end
    end
    uiConnections = {}

    for _, conn in ipairs(globalConns) do
        if conn and conn.Connected then conn:Disconnect() end
    end
    globalConns      = {}
    _playerCharConns = {} 

    for k in pairs(toggles) do
        if k ~= "tpToolObject" then toggles[k] = false end
    end

    if screenGui and screenGui.Parent then screenGui:Destroy() end
    _refreshColorsPending = false
    _G.UniversalScript = nil
end
